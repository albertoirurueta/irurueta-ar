27  /**
28   * Compares two fundamental matrices by estimating average epipolar distances.
29   * This class uses epipolar geometry to determine how similar fundamental 
30   * matrices are. This is done by assuming a certain size on the retinal planes
31   * and generating random 2D points to obtain their associated epipolar lines and
32   * the distances of matched points to those epipolar lines
33   * This class simply computes the norm of the difference of both fundamental
34   * matrices. The smaller the value the more similar the fundamental matrices
35   * will be from a pure algebraic point of view.
36   */
37  @SuppressWarnings({"WeakerAccess", "Duplicates"})
38  public class EpipolarDistanceFundamentalMatrixComparator extends 
39          FundamentalMatrixComparator {

531     /**
532      * Compares two fundamental matrices and returns the comparison value.
533      * Comparison value will depend on the method implemented to compare both
534      * fundamental matrices.
535      * @return comparison value. Typically the smaller the absolute value the
536      * more similar the fundamental matrices are.
537      * @throws NotReadyException if this comparator is not  yet ready to start
538      * the comparison.
539      * @throws LockedException if this instance is locked.
540      * @throws FundamentalMatrixComparatorException if comparison fails due to
541      * some other reason.
542      */    
543     @Override
544     public double compare() throws NotReadyException, LockedException, 
545             FundamentalMatrixComparatorException {
546         if (isLocked()) {
547             throw new LockedException();
548         }
549         if (!isReady()) {
550             throw new NotReadyException();
551         }
552                 
553         try {
554             mLocked = true;            
555             
556             if (mListener != null) {
557                 mListener.onCompareStart(this);            
558             }                        
559             
560             double minHorizontalDisparity = mMinHorizontalDisparityFactor * 
561                     (mMaxX - mMinX);
562             double maxHorizontalDisparity = mMaxHorizontalDisparityFactor *
563                     (mMaxX - mMinX);
564             double minVerticalDisparity = mMinVerticalDisparityFactor *
565                     (mMaxY - mMinY);
566             double maxVerticalDisparity = mMaxVerticalDisparityFactor *
567                     (mMaxY - mMinY);
568             
569             double d1, d1prime, d2, d2prime;
570             double inhomX, inhomY;
571             boolean repeat;
572             int counter;
573             Point2D m1 = Point2D.create(
574                     CoordinatesType.HOMOGENEOUS_COORDINATES);
575             Point2D m2 = Point2D.create(
576                     CoordinatesType.HOMOGENEOUS_COORDINATES);
577             Line2D l1real = new Line2D();
578             Line2D l1est = new Line2D();
579             Line2D l2real = new Line2D();
580             Line2D l2est = new Line2D();
581             UniformRandomizer randomizer = new UniformRandomizer(new Random());
582             double avgDist = 0.0;
583             int maxIterations = (int)(((double)mNSamples) * mMaxIterationsFactor);
584             int currentIter = 0;            
585             float progress, previousProgress = 0.0f;
586             for (int i = 0; i < mNSamples; i++) {
587                 repeat = true;
588                 counter = 0;
589                 while (repeat && (counter < mNSamples)) {
590                     //set x value
591                     inhomX = randomizer.nextDouble(mMinX, mMaxX);
592                     
593                     //set y value
594                     inhomY = randomizer.nextDouble(mMinY, mMaxY);
595                     m1.setInhomogeneousCoordinates(inhomX, inhomY);
596                     m1.normalize();
597                     
598                     //real epipolar line for random point m1
599                     mGroundTruthFundamentalMatrix.rightEpipolarLine(m1, l2real);
600                     l2real.normalize();
601                     
602                     //check that epipolar line lies within retinal plane size
603                     //taking into account the following equation
604                     //x*l2.getA() + y*l2.getB() + l2.getC() = 0
605                     double yMinX, yMaxX;
606                     if (Math.abs(l2real.getB()) > Double.MIN_VALUE) {
607                         //for x = mMinX
608                         yMinX = -(mMinX * l2real.getA() + l2real.getC()) /
609                                 l2real.getB();
610                         
611                         //for x = mMaxX
612                         yMaxX = -(mMaxX * l2real.getA() + l2real.getC()) /
613                                 l2real.getB();
614                     } else {
615                         yMinX = l2real.getA() >= 0.0 ?
616                                 -Double.MAX_VALUE : Double.MAX_VALUE;
617                         yMaxX = -yMinX;
618                     }
619                     
620                     //for y = mMinY
621                     double xMinY = -(mMinY * l2real.getB() + l2real.getC()) /
622                             l2real.getA();
623                     
624                     //for y = mMaxY
625                     double xMaxY = -(mMaxY * l2real.getB() + l2real.getC()) /
626                             l2real.getA();
627                     
628                     //if epipolar line does not intersect second image, we need
629                     //to repeat with a different sample
630                     repeat = (((yMinX < mMinY) && (yMaxX < mMinY)) || ((yMinX > mMaxY) &&
631                         (yMaxX > mMaxY)) || ((xMinY < mMinX) && (xMaxY < mMinX)) ||
632                         ((xMinY > mMaxX) && (xMaxY > mMaxX)));
633                     counter++;
634                     currentIter++;
635                 }
636                 
637                 if (counter >= mNSamples) {
638                     continue;
639                 }
640                 
641                 //choose point lying on epipolar line l2real :
642                 //m2.getX() * l2real.getA() + m2.getY() * l2real.getB() +
643                 //l2real.getC() = 0
644                 //choose random horizontal component within provided disparity:
645                 inhomX = m1.getInhomX() + randomizer.nextDouble(
646                         minHorizontalDisparity, maxHorizontalDisparity);
647                 if (Math.abs(l2real.getB()) > Double.MIN_VALUE) {
648                     inhomY = -(inhomX * l2real.getA() + l2real.getC()) /
649                             l2real.getB();
650                 } else {
651                     inhomY = Double.MAX_VALUE;
652                 }
653                 
654                 //if point lies outside retinal plane limits, try setting random
655                 //vertical component within provided disparity
656                 if ((inhomY < mMinY) || (inhomY > mMaxY)) {
657                     inhomY = m1.getInhomY() + randomizer.nextDouble(
658                             minVerticalDisparity, maxVerticalDisparity);
659                     inhomX = -(inhomY * l2real.getB() + l2real.getC()) /
660                             l2real.getA();
661                 }
662                 
663                 m2.setInhomogeneousCoordinates(inhomX, inhomY);
664                 m2.normalize();
665                 
666                 //estimated epipolar line for some random point m1
667                 mOtherFundamentalMatrix.rightEpipolarLine(m1, l2est);
668                 l2est.normalize();
669                 
670                 //compute distance from l2est to m2 (distance from estimated to
671                 //real)
672                 d1prime = Math.abs(l2est.signedDistance(m2));
673                 
674                 mOtherFundamentalMatrix.leftEpipolarLine(m2, l1est);
675                 l1est.normalize();
676                 d1 = Math.abs(l1est.signedDistance(m1));
677                 
678                 //repeat reversing roles of ground truth and other fundamental
679                 //matrix
680                 repeat = true;
681                 counter = 0;
682                 while (repeat && (counter < mNSamples)) {
683                     //set x value
684                     inhomX = randomizer.nextDouble(mMinX, mMaxX);
685                     
686                     //set y value
687                     inhomY = randomizer.nextDouble(mMinY, mMaxY);
688                     m1.setInhomogeneousCoordinates(inhomX, inhomY);
689                     m1.normalize();
690                     
691                     //real epipolar line for random point m1
692                     mOtherFundamentalMatrix.rightEpipolarLine(m1, l2est);
693                     l2est.normalize();
694                     
695                     //check that epipolar line lies within retinal plane size
696                     //taking into account the following equation
697                     //x*l2.getA() + y*l2.getB() + l2.getC() = 0
698                     double yMinX, yMaxX;
699                     if (Math.abs(l2est.getB()) > Double.MIN_VALUE) {
700                         //for x = mMinX
701                         yMinX = -(mMinX * l2est.getA() + l2est.getC()) /
702                                 l2est.getB();
703                         
704                         //for x = mMaxX
705                         yMaxX = -(mMaxX * l2est.getA() + l2est.getC()) /
706                                 l2est.getB();
707                     } else {
708                         yMinX = l2est.getA() >= 0.0 ? -Double.MAX_VALUE : 
709                                 Double.MAX_VALUE;
710                         yMaxX = -yMinX;
711                     }
712                     
713                     //for y = mMinY
714                     double xMinY = -(mMinY * l2est.getB() + l2est.getC()) /
715                             l2est.getA();
716                     
717                     //for y = mMaxY
718                     double xMaxY = -(mMaxY * l2est.getB() + l2est.getC()) /
719                             l2est.getA();
720                     
721                     //if epipolar line does not intersect second image, we need
722                     //to repeat with a different sample
723                     repeat = (((yMinX < mMinY) && (yMaxX < mMinY)) || ((yMinX > mMaxY) &&
724                         (yMaxX > mMaxY)) || ((xMinY < mMinX) && (xMaxY < mMinX)) ||
725                     ((xMinY > mMaxX) && (xMaxY > mMaxX)));
726                     counter++;
727                     currentIter++;
728                 }
729                 
730                 if (counter >= mNSamples) {
731                     continue;
732                 }
733                 
734                 //choose point lying on epipolar line l2est :
735                 //m2.getX() * l2est.getA() + m2.getY() * l2est.getB() +
736                 //l2est.getC() = 0
737                 //choose random horizontal component within provided disparity:
738                 inhomX = m1.getInhomX() + randomizer.nextDouble(
739                         minHorizontalDisparity, maxHorizontalDisparity);
740                 if (Math.abs(l2real.getB()) > Double.MIN_VALUE) {
741                     inhomY = -(inhomX * l2est.getA() + l2est.getC()) /
742                             l2est.getB();
743                 } else {
744                     inhomY = Double.MAX_VALUE;
745                 }
746                 
747                 //if point lies outside retinal plane limits, try setting random
748                 //Vertical component within provided disparity
749                 if ((inhomY < mMinY) || (inhomY > mMaxY)) {
750                     inhomY = m1.getInhomY() + randomizer.nextDouble(
751                             minVerticalDisparity, maxVerticalDisparity);
752                     inhomX = -(inhomY * l2est.getB() + l2est.getC()) /
753                             l2est.getA();
754                 }
755                 
756                 m2.setInhomogeneousCoordinates(inhomX, inhomY);
757                 m2.normalize();
758                 
759                 //estimated epipolar line for some random point m1
760                 mOtherFundamentalMatrix.rightEpipolarLine(m1, l2real);
761                 l2real.normalize();
762                 
763                 //compute distance from l2real to m2 (distance from estimated to
764                 //real)
765                 d2prime = Math.abs(l2real.signedDistance(m2));
766                 
767                 mOtherFundamentalMatrix.leftEpipolarLine(m2, l1real);
768                 l1real.normalize();
769                 d2 = Math.abs(l1real.signedDistance(m1));
770                 
771                 avgDist += (d1 + d1prime + d2 + d2prime) / 4.0;
772                 
773                 if (currentIter > maxIterations) {
774                     throw new FundamentalMatrixComparatorException();
775                 }
776                 
777                 progress = (float)currentIter / (float)maxIterations;
778                 
779                 if (mListener != null) {
780                     if (progress - previousProgress > mProgressDelta) {
781                         previousProgress = progress;
782                         mListener.onCompareProgressChange(this, progress);
783                     }
784                 }
785             }
786             
787             
788             if (mListener != null) {
789                 mListener.onCompareEnd(this);
790             }
791         
792             return avgDist / (double)mNSamples;
793             
794         } finally {
795             mLocked = false;
796         }                  
797     }

805         return FundamentalMatrixComparatorType.EPIPOLAR_DISTANCE_COMPARATOR;
806     }
