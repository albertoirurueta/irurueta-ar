35  /**
36   * Non-robust fundamental matrix estimator that uses 7 matched 2D points on
37   * left and right views.
38   * Although this algorithm requires one less point than the 8 points algorithm,
39   * it might return up to three different fundamental matrices that must be 
40   * later checked (i.e. using a robust algorithm to maximize inliers) so that
41   * the correct solution is picked.
42   * If the correct solution is found, it typically obtains results with higher
43   * accuracy than the 8 points algorithm, because rank-2 is not approximated
44   * using SVD, and rather it is accurately enforced.
45   */
46  @SuppressWarnings({"WeakerAccess", "Duplicates"})
47  public class SevenPointsFundamentalMatrixEstimator extends 
48          FundamentalMatrixEstimator {
49      
50      /**
51       * Constant indicating that by default an LMSE solution is not allowed.
52       */
53      public static final boolean DEFAULT_ALLOW_LMSE_SOLUTION = false;
54      
55      /**
56       * Minimum number of matched 2D points to start the estimation.
57       */
58      public static final int MIN_REQUIRED_POINTS = 7;
59      
60      /**
61       * Indicates if by default provided point correspondences are normalized to
62       * increase the accuracy of the estimation.
63       */
64      public static final boolean DEFAULT_NORMALIZE_POINT_CORRESPONDENCES = true;        
65      
66      /**
67       * Tiniest value closest to zero.
68       */
69      public static final double EPS = Double.MIN_VALUE;
70      
71      /**
72       * Indicates whether an LMSE (Least Mean Square Error) solution is allowed
73       * or not. When an LMSE solution is allowed, more than 7 matched points can
74       * be used for fundamental matrix estimation. If LMSE solution is not 
75       * allowed then only the 7 former matched points will be taken into account.
76       */
77      private boolean mAllowLMSESolution;
78      
79      /**
80       * Indicates whether provided matched 2D points must be normalized to 
81       * increase the accuracy of the estimation.
82       */
83      private boolean mNormalizePoints;
84      
85      /**
86       * Constructor.
87       */
88      public SevenPointsFundamentalMatrixEstimator() {
89          super();
90          mAllowLMSESolution = DEFAULT_ALLOW_LMSE_SOLUTION;
91          mNormalizePoints = DEFAULT_NORMALIZE_POINT_CORRESPONDENCES;
92      }
93      
94      /**
95       * Constructor with matched 2D points.
96       * @param leftPoints 2D points on left view.
97       * @param rightPoints 2D points on right view.
98       * @throws IllegalArgumentException if provided list of points do not
99       * have the same length.
100      */
101     public SevenPointsFundamentalMatrixEstimator(List<Point2D> leftPoints, 
102             List<Point2D> rightPoints) {
103         super(leftPoints, rightPoints);
104         mAllowLMSESolution = DEFAULT_ALLOW_LMSE_SOLUTION;
105         mNormalizePoints = DEFAULT_NORMALIZE_POINT_CORRESPONDENCES;
106     }
107 
108     /**
109      * Returns boolean indicating whether an LMSE (Least Mean Square Error) 
110      * solution is allowed or not. When an LMSE solution is allowed, more than 8
111      * matched points can be used for fundamental matrix estimation. If LMSE 
112      * solution is not allowed then only the 7 former matched points will be 
113      * taken into account.
114      * @return true if an LMSE solution is allowed, false otherwise.
115      */
116     public boolean isLMSESolutionAllowed() {
117         return mAllowLMSESolution;
118     }
119     
120     /**
121      * Sets boolean indicating whether an LMSE (LEast Mean Square Error) 
122      * solution is allowed or not. When an LMSE solution is allowed, more than 8 
123      * matched points can be used for fundamental matrix estimation. If LMSE 
124      * solution is not allowed then only the 7 former matched points will be 
125      * taken into account.
126      * @param allowed true if an LMSE solution is allowed, false otherwise.
127      * @throws LockedException if this instance is locked because an estimation
128      * is in progress.
129      */
130     public void setLMSESolutionAllowed(boolean allowed) throws LockedException {
131         if (isLocked()) {
132             throw new LockedException();
133         }
134         
135         mAllowLMSESolution = allowed;
136     }
137 
138     /**
139      * Indicates whether provided matched 2D points must be normalized to 
140      * increase the accuracy of the estimation.
141      * @return true if points must be normalized, false otherwise.
142      */
143     public boolean arePointsNormalized() {
144         return mNormalizePoints;
145     }
146     
147     /**
148      * Sets boolean indicating whether provided matched 2D points must be
149      * normalized to increase the accuracy of the estimation.
150      * @param normalizePoints true if points must be normalized, false 
151      * otherwise.
152      * @throws LockedException if this instance is locked because an estimation
153      * is in progress.
154      */
155     public void setPointsNormalized(boolean normalizePoints) 
156             throws LockedException {
157         if (isLocked()) {
158             throw new LockedException();
159         }
160         
161         mNormalizePoints = normalizePoints;
162     }
163     
164     /**
165      * Returns boolean indicating whether estimator is ready to start the 
166      * fundamental matrix estimation.
167      * This is true when the required minimum number of matched points is 
168      * provided to obtain a solution and both left and right views have the
169      * same number of matched points.
170      * @return true if estimator is ready to start the fundamental matrix
171      * estimation, false otherwise.
172      */    
173     @Override
174     public boolean isReady() {
175         return mLeftPoints != null && mRightPoints != null &&
176                 mLeftPoints.size() == mRightPoints.size() &&
177                 mLeftPoints.size() >= MIN_REQUIRED_POINTS;
178     }
179     
180     /**
181      * Estimates all possible fundamental matrices found using provided points.
182      * Because the algorithm uses 7 points and enforces rank 2 by solving a
183      * third degree polynomial, the algorithm might return 1 real solution
184      * (and 2 imaginary ones which are discarded), 3 real solutions, or 1 real
185      * solution with triple multiplicity.
186      * @return all posible fundamental matrices found using provided points.
187      * @throws LockedException if estimator is locked doing an estimation.
188      * @throws NotReadyException if estimator is not ready because required 
189      * input points have not already been provided.
190      * @throws FundamentalMatrixEstimatorException if configuration of provided
191      * 2D points is degenerate and fundamental matrix estimation fails.
192      */
193     public List<FundamentalMatrix> estimateAll() throws LockedException,
194             NotReadyException, FundamentalMatrixEstimatorException {
195         List<FundamentalMatrix> result = new ArrayList<>();
196         estimateAll(result);
197         return result;
198     }
199     
200     /**
201      * Estimates all possible fundamental matrices found using provided points
202      * and adds the result to provided result list.
203      * Because the algorithm uses 7 points and enforces rank 2 by solving a
204      * third degree polynomial, the algorithm might return 1 real solution
205      * (and 2 imaginary ones which are discarded), 3 real solutions, or 1 real
206      * solution with triple multiplicity.
207      * @param result list where results will be stored.
208      * @throws LockedException if estimator is locked doing an estimation.
209      * @throws NotReadyException if estimator is not ready because required 
210      * input points have not already been provided.
211      * @throws FundamentalMatrixEstimatorException if configuration of provided
212      * 2D points is degenerate and fundamental matrix estimation fails.
213      */
214     public void estimateAll(List<FundamentalMatrix> result) 
215             throws LockedException, NotReadyException, 
216             FundamentalMatrixEstimatorException {
217         
218         if (isLocked()) {
219             throw new LockedException();
220         }
221         if (!isReady()) {
222             throw new NotReadyException();
223         }
224         
225         mLocked = true;
226         
227         result.clear();
228                 
229         if (mListener != null) {
230             mListener.onEstimateStart(this);
231         }
232         
233         int nPoints = mLeftPoints.size();
234                 
235         try {
236             ProjectiveTransformation2D leftNormalization = null;
237             ProjectiveTransformation2D rightNormalization = null;
238             List<Point2D> leftPoints;
239             List<Point2D> rightPoints;
240             if (mNormalizePoints) {
241                 //normalize points on left view
242                 Point2DNormalizer normalizer = new Point2DNormalizer(
243                         mLeftPoints);
244                 normalizer.compute();                
245                 
246                 leftNormalization = normalizer.getTransformation();
247                 
248                 //normalize points on right view
249                 normalizer.setPoints(mRightPoints);
250                 normalizer.compute();
251                 
252                 rightNormalization = normalizer.getTransformation();
253                 
254                 //normalize to increase accuracy
255                 leftNormalization.normalize();
256                 rightNormalization.normalize();
257                 
258                 leftPoints = leftNormalization.transformPointsAndReturnNew(
259                         mLeftPoints);
260                 rightPoints = rightNormalization.transformPointsAndReturnNew(
261                         mRightPoints);
262             } else {
263                 leftPoints = mLeftPoints;
264                 rightPoints = mRightPoints;
265             }
266             
267             Matrix a;
268             if (isLMSESolutionAllowed()) {
269                 a = new Matrix(nPoints, 9);
270             } else {
271                 a = new Matrix(MIN_REQUIRED_POINTS, 9);
272             }
273             
274             Point2D leftPoint;
275             Point2D rightPoint;
276             double homLeftX;
277             double homLeftY;
278             double homLeftW;
279             double homRightX;
280             double homRightY;
281             double homRightW;
282             double value0;
283             double value1;
284             double value2;
285             double value3;
286             double value4;
287             double value5;
288             double value6;
289             double value7;
290             double value8;
291             double rowNorm;
292             for (int i = 0; i < nPoints; i++) {
293                 leftPoint = leftPoints.get(i);
294                 rightPoint = rightPoints.get(i);
295                 
296                 //normalize points to increase accuracy
297                 leftPoint.normalize();
298                 rightPoint.normalize();
299                                 
300                 homLeftX = leftPoint.getHomX();
301                 homLeftY = leftPoint.getHomY();
302                 homLeftW = leftPoint.getHomW();
303                 
304                 homRightX = rightPoint.getHomX();
305                 homRightY = rightPoint.getHomY();
306                 homRightW = rightPoint.getHomW();
307                 
308                 //set a row values
309                 value0 = homLeftX * homRightX;
310                 value1 = homLeftY * homRightX;
311                 value2 = homLeftW * homRightX;
312 			
313                 value3 = homLeftX * homRightY;
314                 value4 = homLeftY * homRightY;
315                 value5 = homLeftW * homRightY;
316 			
317                 value6 = homLeftX * homRightW;
318                 value7 = homLeftY * homRightW;
319                 value8 = homLeftW * homRightW;
320                 
321                 //normalize row to increase accuracy
322                 rowNorm = Math.sqrt(Math.pow(value0, 2.0) +
323                         Math.pow(value1, 2.0) + Math.pow(value2, 2.0) +
324                         Math.pow(value3, 2.0) + Math.pow(value4, 2.0) +
325                         Math.pow(value5, 2.0) + Math.pow(value6, 2.0) +
326                         Math.pow(value7, 2.0) + Math.pow(value8, 2.0));
327                 
328                 a.setElementAt(i, 0, value0 / rowNorm);
329                 a.setElementAt(i, 1, value1 / rowNorm);
330                 a.setElementAt(i, 2, value2 / rowNorm);
331                 a.setElementAt(i, 3, value3 / rowNorm);
332                 a.setElementAt(i, 4, value4 / rowNorm);
333                 a.setElementAt(i, 5, value5 / rowNorm);
334                 a.setElementAt(i, 6, value6 / rowNorm);
335                 a.setElementAt(i, 7, value7 / rowNorm);
336                 a.setElementAt(i, 8, value8 / rowNorm);
337 
338                 if(!isLMSESolutionAllowed() && i == (MIN_REQUIRED_POINTS - 1))
339                     break;
340             }
341             
342             SingularValueDecomposer decomposer = new SingularValueDecomposer(a);
343             
344             decomposer.decompose();
345             
346             //having 7 points for 9 variables means that rank can be as high as
347             //7, and nullity must be 2, so that the final fundamental matrix
348             //can be found as a linear combination of the null-space.
349             //If nullity is bigger than 2, then geometry is degenerate, usually
350             //due to colinearities or coplanarities on projected image points.
351             //In this case we throw an exception
352             if (decomposer.getNullity() > 2) {
353                 throw new FundamentalMatrixEstimatorException();
354             }
355             
356             Matrix v = decomposer.getV();
357             
358             //The last two column vectors of V contain the "base" matrices
359             //to be used for the retrieval of the true fundamental matrix, since
360             //the fundamental matrix we are looking for will be a linear 
361             //combination of such matrices after reshaping the vectors into 3x3 
362             //matrices
363             Matrix fundMatrix1 = new Matrix(
364                     FundamentalMatrix.FUNDAMENTAL_MATRIX_ROWS,
365                     FundamentalMatrix.FUNDAMENTAL_MATRIX_COLS);
366             fundMatrix1.setElementAt(0, 0, v.getElementAt(0, 8));
367             fundMatrix1.setElementAt(0, 1, v.getElementAt(1, 8));
368             fundMatrix1.setElementAt(0, 2, v.getElementAt(2, 8));
369             fundMatrix1.setElementAt(1, 0, v.getElementAt(3, 8));
370             fundMatrix1.setElementAt(1, 1, v.getElementAt(4, 8));
371             fundMatrix1.setElementAt(1, 2, v.getElementAt(5, 8));
372             fundMatrix1.setElementAt(2, 0, v.getElementAt(6, 8));
373             fundMatrix1.setElementAt(2, 1, v.getElementAt(7, 8));
374             fundMatrix1.setElementAt(2, 2, v.getElementAt(8, 8));
375 
376             Matrix fundMatrix2 = new Matrix(
377                     FundamentalMatrix.FUNDAMENTAL_MATRIX_ROWS,
378                     FundamentalMatrix.FUNDAMENTAL_MATRIX_COLS);
379             fundMatrix2.setElementAt(0, 0, v.getElementAt(0, 7));
380             fundMatrix2.setElementAt(0, 1, v.getElementAt(1, 7));
381             fundMatrix2.setElementAt(0, 2, v.getElementAt(2, 7));
382             fundMatrix2.setElementAt(1, 0, v.getElementAt(3, 7));
383             fundMatrix2.setElementAt(1, 1, v.getElementAt(4, 7));
384             fundMatrix2.setElementAt(1, 2, v.getElementAt(5, 7));
385             fundMatrix2.setElementAt(2, 0, v.getElementAt(6, 7));
386             fundMatrix2.setElementAt(2, 1, v.getElementAt(7, 7));
387             fundMatrix2.setElementAt(2, 2, v.getElementAt(8, 7));
388             
389             if (mNormalizePoints && leftNormalization != null) {
390                 //denormalize linear combination of fundamental matrices
391                 //fundMatrix1 and fundMatrix2
392                 Matrix transposedRightTransformationMatrix =
393                         rightNormalization.asMatrix().transposeAndReturnNew();
394                 Matrix leftTransformationMatrix = leftNormalization.asMatrix();
395                 
396                 //compute fundMatrix1 = transposedRightTransformationMatrix *
397                 //fundMatrix1 * leftTransformationMatrix
398                 fundMatrix1.multiply(leftTransformationMatrix);
399                 fundMatrix1 = transposedRightTransformationMatrix.
400                         multiplyAndReturnNew(fundMatrix1);
401                 
402                 //normalize by frobenius norm to increase accuracy after point
403                 //denormalization
404                 double norm = Utils.normF(fundMatrix1);
405                 fundMatrix1.multiplyByScalar(1.0 / norm);
406                 
407                 //compute fundMatrix2 = transposedRightTransformationMatrix *
408                 //fundMatrix2 * leftTransformationMatrix
409                 fundMatrix2.multiply(leftTransformationMatrix);
410                 transposedRightTransformationMatrix.multiply(fundMatrix2);
411                 fundMatrix2 = transposedRightTransformationMatrix;
412 
413                 //normalize by frobenius norm to increase accuracy after point
414                 //denormalization
415                 norm = Utils.normF(fundMatrix2);
416                 fundMatrix2.multiplyByScalar(1.0 / norm);                
417             }
418             
419             //because fundMatrix1, and fundMatrix2 have been obtained as 
420             //columns of V, then its frobenius norm will be 1 because SVD 
421             //already returns normalized singular vectors, and there is no need 
422             //to normalize by frobenius norm if points are NOT normalized
423             
424             //The last thing we need to do is to enforce rank 2 on fundamental
425             //matrix, since we know it always is a rank 2 matrix. For that 
426             //reason we know that det(F) = 0.
427             //Since the fundamental matrix F is a linear combination of the
428             //two matrices F1, F2 we have found then: F = b * F1 + (1.0 - b) *F2
429             //where b will range from 0 to 1. 
430             //Hence: det(b * F1 + (1.0 - b) * F2) = 0
431             //This produces a third degree polynomial as follows:
432 
433             //coefficients of polynomial: a*x^3 + b*x^2 + c*x + d
434             double aPoly = 0.0;
435             double bPoly = 0.0;
436             double cPoly = 0.0;
437             double dPoly = 0.0;
438             double[] params = new double[4];
439             
440             computeParams(fundMatrix1.getElementAt(0, 0), 
441                     fundMatrix2.getElementAt(0, 0),
442                     fundMatrix1.getElementAt(1, 1), 
443                     fundMatrix2.getElementAt(1, 1),
444                     fundMatrix1.getElementAt(2, 2), 
445                     fundMatrix2.getElementAt(2, 2), params);
446             
447             aPoly += params[0];
448             bPoly += params[1];
449             cPoly += params[2];
450             dPoly += params[3];
451             
452             computeParams(fundMatrix1.getElementAt(2, 1), 
453                     fundMatrix2.getElementAt(2, 1),
454                     fundMatrix1.getElementAt(1, 0),
455                     fundMatrix2.getElementAt(1, 0),
456                     fundMatrix1.getElementAt(0, 2),
457                     fundMatrix2.getElementAt(0, 2), params);
458 
459             aPoly += params[0];
460             bPoly += params[1];
461             cPoly += params[2];
462             dPoly += params[3];
463             
464             computeParams(fundMatrix1.getElementAt(2, 0),
465                     fundMatrix2.getElementAt(2, 0),
466                     fundMatrix1.getElementAt(0, 1),
467                     fundMatrix2.getElementAt(0, 1),
468                     fundMatrix1.getElementAt(1, 2),
469                     fundMatrix2.getElementAt(1, 2), params);
470             
471             aPoly += params[0];
472             bPoly += params[1];
473             cPoly += params[2];
474             dPoly += params[3];
475             
476             computeParams(fundMatrix1.getElementAt(2, 0), 
477                     fundMatrix2.getElementAt(2, 0),
478                     fundMatrix1.getElementAt(1, 1),
479                     fundMatrix2.getElementAt(1, 1),
480                     fundMatrix1.getElementAt(0, 2),
481                     fundMatrix2.getElementAt(0, 2), params);
482             
483             aPoly -= params[0];
484             bPoly -= params[1];
485             cPoly -= params[2];
486             dPoly -= params[3];
487             
488             computeParams(fundMatrix1.getElementAt(1, 2),
489                     fundMatrix2.getElementAt(1, 2),
490                     fundMatrix1.getElementAt(2, 1),
491                     fundMatrix2.getElementAt(2, 1),
492                     fundMatrix1.getElementAt(0, 0),
493                     fundMatrix2.getElementAt(0, 0), params);
494             
495             aPoly -= params[0];
496             bPoly -= params[1];
497             cPoly -= params[2];
498             dPoly -= params[3];
499             
500             computeParams(fundMatrix1.getElementAt(1, 0),
501                     fundMatrix2.getElementAt(1, 0),
502                     fundMatrix1.getElementAt(0, 1),
503                     fundMatrix2.getElementAt(0, 1),
504                     fundMatrix1.getElementAt(2, 2),
505                     fundMatrix2.getElementAt(2, 2), params);
506             
507             aPoly -= params[0];
508             bPoly -= params[1];
509             cPoly -= params[2];
510             dPoly -= params[3];
511 
512             //normalize polynomial coefficients to increase accuracy
513             double coeffNorm = Math.sqrt(Math.pow(aPoly, 2.0) +
514                     Math.pow(bPoly, 2.0) + Math.pow(cPoly, 2.0) +
515                     Math.pow(dPoly, 2.0));
516             aPoly /= coeffNorm;
517             bPoly /= coeffNorm;
518             cPoly /= coeffNorm;
519             dPoly /= coeffNorm;
520             
521             //store polynomial coefficients into array and find its roots to
522             //enforce det(F) = 0. The solution must be unique and real!
523             params[0] = dPoly;
524             params[1] = cPoly;
525             params[2] = bPoly;
526             params[3] = aPoly;
527             
528             double beta1 = 0.0;
529             double beta2 = 0.0;
530             double beta3 = 0.0;
531             boolean beta1Available = false;
532             boolean beta2Available= false;
533             boolean beta3Available = false;
534             Complex[] roots;
535             Complex root1;
536             Complex root2;
537             Complex root3;
538             
539             if (ThirdDegreePolynomialRootsEstimator.isThirdDegree(params)) {
540                 //solve third degree polynomial
541                 ThirdDegreePolynomialRootsEstimator estimator =
542                         new ThirdDegreePolynomialRootsEstimator(params);
543                 estimator.estimate();
544                 roots = estimator.getRoots();
545                 root1 = roots[0];
546                 root2 = roots[1];
547                 root3 = roots[2];
548                                 
549                 if (Math.abs(root1.getImaginary()) <= EPS) {
550                     //1st root is real, so we keep it
551                     beta1 = root1.getReal();
552                     beta1Available = true;
553                 }
554                 if (Math.abs(root2.getImaginary()) <= EPS) {
555                     //2nd root is real, so we keep it
556                     beta2 = root2.getReal();
557                     beta2Available = true;
558                 }
559                 if (Math.abs(root3.getImaginary()) <= EPS) {
560                     //3rd root is real, so we keep it
561                     beta3 = root3.getReal();
562                     beta3Available = true;
563                 }
564             } else if (SecondDegreePolynomialRootsEstimator.isSecondDegree(
565                     params)) {
566                 //solve second degree polynomial
567                 SecondDegreePolynomialRootsEstimator estimator =
568                         new SecondDegreePolynomialRootsEstimator(params);
569                 estimator.estimate();
570                 roots = estimator.getRoots();
571                 root1 = roots[0];
572                 root2 = roots[1];
573                 if (Math.abs(root1.getImaginary()) <= EPS) {
574                     //1st root is real, so we keep it
575                     beta1 = root1.getReal();
576                     beta1Available = true;
577                 }
578                 if (Math.abs(root2.getImaginary()) <= EPS) {
579                     //2nd root is real, so we keep it
580                     beta2 = root2.getReal();
581                     beta2Available = true;
582                 }
583             } else if (FirstDegreePolynomialRootsEstimator.isFirstDegree(
584                     params)) {
585                 //solve first degree polynomial
586                 FirstDegreePolynomialRootsEstimator estimator =
587                         new FirstDegreePolynomialRootsEstimator(params);
588                 estimator.estimate();
589                 roots = estimator.getRoots();
590                 root1 = roots[0]; //this is the only solution and is real 
591                                 //becuase coefficients are real
592                 beta1 = root1.getReal();
593                 beta1Available = true;
594             } else {
595                 //invalid polynomial degree
596                 throw new FundamentalMatrixEstimatorException();
597             }
598             
599             if (!beta1Available && !beta2Available && !beta3Available) {
600                 //No solution was found
601                 throw new FundamentalMatrixEstimatorException();
602             }
603             
604             //Once the polynomial is solved we compute the linear combination
605             //F = b * F1 + (1 - b) * F2 which has rank 2 using all available 
606             //solutions
607             Matrix fundMatrix;
608             FundamentalMatrix f;
609             result.clear(); //clear previous values
610             if (beta1Available) {
611                 fundMatrix = fundMatrix1.multiplyByScalarAndReturnNew(beta1).
612                         addAndReturnNew(fundMatrix2.
613                                 multiplyByScalarAndReturnNew(1.0 - beta1));
614                 if (enforceRank2(fundMatrix, decomposer)) {
615                     throw new FundamentalMatrixEstimatorException();
616                 }
617                 
618                 f = new FundamentalMatrix(fundMatrix);
619                 result.add(f);
620             }
621             if (beta2Available) {
622                 fundMatrix = fundMatrix1.multiplyByScalarAndReturnNew(beta2).
623                         addAndReturnNew(fundMatrix2.multiplyByScalarAndReturnNew(
624                         1.0 - beta2));
625                 if (enforceRank2(fundMatrix, decomposer)) {
626                     throw new FundamentalMatrixEstimatorException();
627                 }
628                 
629                 f = new FundamentalMatrix(fundMatrix);
630                 result.add(f);
631             }
632             if (beta3Available) {
633                 fundMatrix = fundMatrix1.multiplyByScalarAndReturnNew(beta3).
634                         addAndReturnNew(fundMatrix2.multiplyByScalarAndReturnNew(
635                         1.0 - beta3));
636                 if (enforceRank2(fundMatrix, decomposer)) {
637                     throw new FundamentalMatrixEstimatorException();
638                 }
639                 
640                 f = new FundamentalMatrix(fundMatrix);
641                 result.add(f);
642             }
643             
644             if (result.isEmpty()) {
645                 //no valid solution was found
646                 throw new FundamentalMatrixEstimatorException();
647             }
648                         
649             if (mListener != null) {
650                 mListener.onEstimateEnd(this, result.get(0));
651             }
652             
653         } catch (InvalidFundamentalMatrixException | AlgebraException |
654                 NumericalException | NormalizerException e) {
655             throw new FundamentalMatrixEstimatorException(e);
656         } finally {
657             mLocked = false;
658         }        
659     }

661     /**
662      * Estimates a fundamental matrix using provided lists of matched points on
663      * left and right views.
664      * This method returns a solution only if one possible solution exists.
665      * If more than one solution is available, this method will fail.
666      * Because this algorithm might return more than one solution, it is highly
667      * encouraged to use estimateAll method instead.
668      * @return a fundamental matrix.
669      * @throws LockedException if estimator is locked doing an estimation.
670      * @throws NotReadyException if estimator is not ready because required 
671      * input points have not already been provided.
672      * @throws FundamentalMatrixEstimatorException if configuration of provided
673      * 2D points is degenerate and fundamental matrix estimation fails or more
674      * than one solution exists.
675      */    
676     @Override
677     public FundamentalMatrix estimate() throws LockedException, 
678             NotReadyException, FundamentalMatrixEstimatorException {
679         List<FundamentalMatrix> list = estimateAll();
680         if (list.size() > 1) {
681             throw new FundamentalMatrixEstimatorException();
682         }
683         
684         return list.get(0);
685     }

707     /**
708      * Enforces rank 2 into provided matrix.
709      * This method modifies provided matrix.
710      * @param matrix matrix to be enforced to have rank 2.
711      * @param decomposer an SVD decomposer.
712      * @return false if rank was successfully enforced, true otherwise.
713      * @throws AlgebraException if an error occurs during SVD decomposition
714      * because of numerical instabilities.
715      */
716     private boolean enforceRank2(Matrix matrix, 
717             SingularValueDecomposer decomposer) throws AlgebraException {
718         
719         decomposer.setInputMatrix(matrix);
720         decomposer.decompose();
721         
722         int rank = decomposer.getRank();
723         if (rank > FundamentalMatrix.FUNDAMENTAL_MATRIX_RANK) {
724             //rank needs to be reduced
725             Matrix u = decomposer.getU();
726             Matrix w = decomposer.getW();
727             Matrix v = decomposer.getV();
728                 
729             //transpose V
730             v.transpose();
731 
732             //set last singular value to zero to enforce rank 2
733             w.setElementAt(2, 2, 0.0);
734                 
735             //compute matrix = U * W * V'
736             w.multiply(v);
737             u.multiply(w);
738             matrix.copyFrom(u);
739             return false;
740         } else {
741             //if rank is 2, rank is ok, otherwise rank is lower than fundamental 
742             //matrix rank (rank 1) and estimation has failed because of 
743             //coplanarities
744             return rank != FundamentalMatrix.FUNDAMENTAL_MATRIX_RANK;
745         }                
746     }

748     /**
749      * Computes parameters of third degree polynomial to obtain possible 
750      * solutions.
751      * @param a 1st param.
752      * @param b 2nd param.
753      * @param c 3rd param.
754      * @param d 4th param.
755      * @param e 5th param.
756      * @param f 6th param.
757      * @param out array of length 4 where partial parameters of third degree
758      * polynomial are stored.
759      */
760     private void computeParams(double a, double b, double c, double d, double e, 
761             double f, double[] out) {
762         
763         double ace = a*c*e;
764         double ade = a*d*e;
765         double bce = b*c*e;
766         double bde = b*d*e;
767         double acf = a*c*f;
768         double adf = a*d*f;
769         double bcf = b*c*f;
770         double bdf = b*d*f;
771         
772         out[0] = ace - ade - bce + bde - acf + adf + bcf - bdf;
773         out[1] = ade + bce - 2.0*bde + acf - 2.0*adf - 2.0*bcf + 3.0*bdf;
774         out[2] = bde + adf + bcf -3.0*bdf;
775         out[3] = bdf;        
776     }
