28  /**
29   * This class defines the interface for an estimator of the Dual Absolute 
30   * Quadric (DAQ) assuming equal vertical and horizontal focal length, no
31   * skewness and principal point at the center of the image. 
32   * The projection of the Dual Absolute Quadric through a given
33   * pinhole camera P, becomes the Dual Image of Absolute Conic (DIAC).
34   * Since the DIAC is directly related to the intrinsic parameters K of the 
35   * camera P used for projection, then by using a pair of arbitrary cameras,
36   * it is possible to solve their corresponding focal length, assuming that:
37   * - cameras are arbitrary (usually the initial camera is the identity and must
38   * be discarded) as it creates a numerical degeneracy.
39   * - all provided cameras have the same intrinsic parameters.
40   * - it is assumed that skewness is zero, the principal point is at the center
41   * of the image plane (zero), and both horizontal and vertical focal planes are
42   * equal.
43   */

45  public abstract class DualAbsoluteQuadricEstimator {

532             throws IllegalArgumentException, LockedException {
533         if (isLocked()) {

535         }

537                 cameras.size() < getMinNumberOfRequiredCameras()) {
538             throw new IllegalArgumentException();
539         }

543     /**
544      * Returns minimum number of required cameras needed to estimate the 
545      * Dual Absolute Quadric (DAQ).
546      * At least 8 equations are needed to solve the DAQ
547      * For each imposed constraint, one less equation is required.
548      * Depending on the number of constraints more or less cameras will be 
549      * required.
550      * If zero skewness is enforced, a solution is available with 8 cameras.
551      * If zero skewness and focal distance aspect ratio is known, then a
552      * solution is available with 4 cameras.
553      * If principal point is located at origin, then a solution is available
554      * with 4 cameras.
555      * If zero skewness and principal point at origin are enforced, then a 
556      * solution is available with 3 cameras.
557      * If zero skewness is enforced, focal distance aspect ratio is known and
558      * principal point is at origin, then a solution is available with 2 
559      * cameras.
560      * NOTE: minimum number of cameras considers only the cameras providing
561      * additional information. If a camera is equivalent to another one or does
562      * not provide additional information (such as a camera at the origin with 
563      * no rotation), then more cameras will be needed.
564      * @return minimum number of required cameras needed to estimate the Dual
565      * Absolute Quadric (DAQ) or -1 if constraints configurations is not valid.
566      */
567     public int getMinNumberOfRequiredCameras() {
568         int numEquations = 0;
569         if (mPrincipalPointAtOrigin) {
570             numEquations += 2;
571             if (mZeroSkewness) {
572                 numEquations++;
573                 
574                 //only a linear solution exists for known aspect ratio if 
575                 //skewness is zero
576                 if (mFocalDistanceAspectRatioKnown) {
577                     numEquations++;
578                 }                
579             }
580         }
581         
582         if (numEquations == 0) {
583             return -1;
584         }
585         
586         int minRequiredEquations = MIN_REQUIRED_EQUATIONS;
587         if (mSingularityEnforced) {
588             minRequiredEquations--;
589         }
590         return (minRequiredEquations / numEquations) +
591                 ((minRequiredEquations % numEquations != 0) ? 1 : 0);
592     }

594     /**
595      * Indicates whether current constraints are enough to start the estimation.
596      * In order to obtain a linear solution for the DAQ estimation, we need at
597      * least the principal point at origin constraint.
598      * @return true if constraints are valid, false otherwise.
599      */
600     public boolean areValidConstraints() {
601         return mPrincipalPointAtOrigin && 
602                 (mFocalDistanceAspectRatioKnown || !mSingularityEnforced);
603     }

612                 mCameras.size() >= getMinNumberOfRequiredCameras() &&
613                 areValidConstraints();

791     /**
792      * Normalizes i-th row of provided matrix a.
793      * @param a matrix whose row must be normalized.
794      * @param i row to be normalized.
795      */
796     protected void normalizeRow(Matrix a, int i) {
797         double rowNorm = 0.0;
798         int cols = a.getColumns();
799         for (int j = 0; j < cols; j++) {
800             rowNorm += Math.pow(a.getElementAt(i, j), 2.0);
801         }
802         
803         rowNorm = Math.sqrt(rowNorm);
804         
805         for (int j = 0; j < cols; j++) {
806             a.setElementAt(i, j, a.getElementAt(i, j) / rowNorm);
807         }
808     }

944     /**
945      * Enforces (if needed) rank 3 of estimated quadric by building a polynomial 
946      * out of the last columns of the singular value vector matrix to obtain a 
947      * linear combination solution.
948      * @param decomposer decomposer containing possible solutions after 
949      * decomsition.
950      * @param result instance where estimated Dual Absolute Quadrics (DAQs) with 
951      * rank 3 enforced will be stored.
952      * @throws AlgebraException if there are numerical instabilities
953      * @throws NumericalException if a solution to the polynomial enforcing
954      * rank 3 cannot be found.
955      * @throws DualAbsoluteQuadricEstimatorException if something else fails.
956      */
957     protected void enforceRank3IfNeeded(
958             SingularValueDecomposer decomposer, 
959             DualAbsoluteQuadric result) throws AlgebraException, 
960             NumericalException, DualAbsoluteQuadricEstimatorException {
961 
962         decomposer.decompose();
963         
964         if (mSingularityEnforced) {
965             if (decomposer.getNullity() > 2) {
966                 //provided cameras are degenerate and there is not a single 
967                 //solution for the DAQ (up to scale)
968                 throw new DualAbsoluteQuadricEstimatorException();                
969             }
970             
971             Matrix v = decomposer.getV();
972             
973             //last 2 columns of v contains parameters parameters a, b, c, d, e, 
974             //f, g, h, i  that can be obtained as a linear combination of
975             //those last columns (i.e. a = a1 + alpha*a2).
976             //By enforcing determinant zero, we can obtain a 4th degree 
977             //polynomial to determine alpha to find the unique linear 
978             //combination of last 2 columns that solve both DAQ equation and 
979             //generates rank 3 DAQ.
980             double a1 = v.getElementAt(0, 8);
981             double b1 = v.getElementAt(1, 8);
982             double c1 = v.getElementAt(2, 8);
983             double d1 = v.getElementAt(3, 8);
984             double e1 = v.getElementAt(4, 8);
985             double f1 = v.getElementAt(5, 8);
986             double g1 = v.getElementAt(6, 8);
987             double h1 = v.getElementAt(7, 8);
988             double i1 = v.getElementAt(8, 8);
989             double j1 = v.getElementAt(9, 8);
990             
991             double a2 = v.getElementAt(0, 9);
992             double b2 = v.getElementAt(1, 9);
993             double c2 = v.getElementAt(2, 9);
994             double d2 = v.getElementAt(3, 9);
995             double e2 = v.getElementAt(4, 9);
996             double f2 = v.getElementAt(5, 9);
997             double g2 = v.getElementAt(6, 9);
998             double h2 = v.getElementAt(7, 9);
999             double i2 = v.getElementAt(8, 9);
1000             double j2 = v.getElementAt(9, 9);
1001             
1002             Polynomial poly = buildPolynomialToEnforceRank3(a1, b1, c1, d1, e1, 
1003                     f1, g1, h1, i1, j1, a2, b2, c2, d2, e2, f2, g2, h2, i2, j2);
1004             
1005             Complex[] roots = poly.getRoots();
1006 
1007             if (roots != null) {
1008                 //pick best solution (closest real root) = evaluation closest to 
1009                 //zero
1010                 double polyEval, bestPolyEval = Double.MAX_VALUE;
1011                 for (Complex root : roots) {
1012                     double real = root.getReal();
1013                     polyEval = Math.abs(poly.evaluate(real));
1014                     if (polyEval < bestPolyEval) {
1015                         bestPolyEval = polyEval;
1016                         double a = a1 + real * a2;
1017                         double b = b1 + real * b2;
1018                         double c = c1 + real * c2;
1019                         double d = d1 + real * d2;
1020                         double e = e1 + real * e2;
1021                         double f = f1 + real * f2;
1022                         double g = g1 + real * g2;
1023                         double h = h1 + real * h2;
1024                         double i = i1 + real * i2;
1025                         double j = j1 + real * j2;
1026                         result.setParameters(a, b, c, d, e, f, g, h, i, j);
1027                     }
1028                 }    
1029             } else {                
1030                 //if no roots could be found, it might be due to numerical
1031                 //inaccuracies, so we find minimum or maximum of polynomial
1032                 //which evaluates closest to zero
1033                 double[] extrema = poly.getExtrema();
1034                 if (extrema == null) {
1035                     //polynomial has no extrema, which means it is degenerate
1036                     throw new DualAbsoluteQuadricEstimatorException();
1037                 }
1038                 
1039                 double polyEval, bestPolyEval = Double.MAX_VALUE;
1040                 for (double extremum : extrema) {
1041                     polyEval = Math.abs(poly.evaluate(extremum));
1042                     if (polyEval < bestPolyEval) {
1043                         bestPolyEval = polyEval;
1044                         double a = a1 + extremum * a2;
1045                         double b = b1 + extremum * b2;
1046                         double c = c1 + extremum * c2;
1047                         double d = d1 + extremum * d2;
1048                         double e = e1 + extremum * e2;
1049                         double f = f1 + extremum * f2;
1050                         double g = g1 + extremum * g2;
1051                         double h = h1 + extremum * h2;
1052                         double i = i1 + extremum * i2;
1053                         double j = j1 + extremum * j2;
1054                         result.setParameters(a, b, c, d, e, f, g, h, i, j);                        
1055                     }
1056                 }
1057                 
1058                 if (mValidateEnforcedSingularity) {
1059                     //check that determinant of estimated DAQ is below allowed 
1060                     //threshold
1061                     double absDet = Math.abs(Utils.det(result.asMatrix()));
1062                     if (absDet > mDeterminantThreshold) {
1063                         //DAQ does not have rank 3
1064                         throw new DualAbsoluteQuadricEstimatorException();
1065                     }
1066                 }
1067             }
1068                         
1069         } else {
1070             if (decomposer.getNullity() > 1) {
1071                 //provided cameras are degenerate and there is not a single 
1072                 //solution for the DAQ (up to scale)
1073                 throw new DualAbsoluteQuadricEstimatorException();
1074             }
1075             
1076             Matrix v = decomposer.getV();
1077             
1078             //last column of v contains parameters a, b, c, d, e, f, g, h, i 
1079             //defining the Dual Absolute Quadric (DAQ) as:
1080             double a = v.getElementAt(0, 9);
1081             double b = v.getElementAt(1, 9);
1082             double c = v.getElementAt(2, 9);
1083             double d = v.getElementAt(3, 9);
1084             double e = v.getElementAt(4, 9);
1085             double f = v.getElementAt(5, 9);
1086             double g = v.getElementAt(6, 9);
1087             double h = v.getElementAt(7, 9);
1088             double i = v.getElementAt(8, 9);
1089             double j = v.getElementAt(9, 9);
1090                 
1091             result.setParameters(a, b, c, d, e, f, g, h, i, j);
1092         }
1093     }

1095     /**
1096      * Builds polynomial to enforce rank 3 (zero determinant).
1097      * @param a1 a of 1st DAQ.
1098      * @param b1 b of 1st DAQ.
1099      * @param c1 c of 1st DAQ.
1100      * @param d1 d of 1st DAQ.
1101      * @param e1 e of 1st DAQ.
1102      * @param f1 f of 1st DAQ.
1103      * @param g1 g of 1st DAQ.
1104      * @param h1 h of 1st DAQ.
1105      * @param i1 i of 1st DAQ.
1106      * @param j1 j of 1st DAQ.
1107      * @param a2 a of 2nd DAQ.
1108      * @param b2 b of 2nd DAQ.
1109      * @param c2 c of 2nd DAQ.
1110      * @param d2 d of 2nd DAQ.
1111      * @param e2 e of 2nd DAQ.
1112      * @param f2 f of 2nd DAQ.
1113      * @param g2 g of 2nd DAQ.
1114      * @param h2 h of 2nd DAQ.
1115      * @param i2 i of 2nd DAQ.
1116      * @param j2 j of 2nd DAQ.
1117      * @return polynomial.
1118      */
1119     private Polynomial buildPolynomialToEnforceRank3(double a1, double b1, 
1120             double c1, double d1, double e1, double f1, double g1, double h1, 
1121             double i1, double j1, double a2, double b2, double c2, double d2, 
1122             double e2, double f2, double g2, double h2, double i2, double j2) {
1123         
1124         Polynomial polyA = new Polynomial(a1, a2);
1125         Polynomial polyB = new Polynomial(b1, b2);
1126         Polynomial polyC = new Polynomial(c1, c2);
1127         Polynomial polyD = new Polynomial(d1, d2);
1128         Polynomial polyE = new Polynomial(e1, e2);
1129         Polynomial polyF = new Polynomial(f1, f2);
1130         Polynomial polyG = new Polynomial(g1, g2);
1131         Polynomial polyH = new Polynomial(h1, h2);
1132         Polynomial polyI = new Polynomial(i1, i2);
1133         Polynomial polyJ = new Polynomial(j1, j2);
1134         
1135         Polynomial result = new Polynomial(5);
1136         
1137         //(a1 + x*a2) * (b1 + x*b2) * (c1 + x*c2) * (j1 + x*j2)
1138         Polynomial tmp = polyA.multiplyAndReturnNew(polyB);
1139         tmp.multiply(polyC);
1140         tmp.multiply(polyJ);
1141         
1142         result.add(tmp);
1143         
1144         //2 * (a1 + x*a2) * (e1 + x*e2) * (h1 + x*h2) * (i1 + x*i2)
1145         tmp = polyA.multiplyAndReturnNew(polyE);
1146         tmp.multiply(polyH);
1147         tmp.multiply(polyI);
1148         tmp.multiplyByScalar(2.0);
1149         
1150         result.add(tmp);
1151         
1152         //-1 * (a1 + x*a2) * (c1 + x*c2) * (h1 + x*h2)^2
1153         tmp = polyA.multiplyAndReturnNew(polyC);
1154         tmp.multiply(polyH);
1155         tmp.multiply(polyH);
1156         tmp.multiplyByScalar(-1.0);
1157         
1158         result.add(tmp);
1159         
1160         //-1 * (a1 + x*a2) * (b1 + x*b2) * (i1 + x*i2)^2
1161         tmp = polyA.multiplyAndReturnNew(polyB);
1162         tmp.multiply(polyI);
1163         tmp.multiply(polyI);
1164         tmp.multiplyByScalar(-1.0);
1165         
1166         result.add(tmp);
1167         
1168         //-1 * (a1 + x*a2) * (e1 + x*e2)^2 * (j1 + x*j2)
1169         tmp = polyA.multiplyAndReturnNew(polyE);
1170         tmp.multiply(polyE);
1171         tmp.multiply(polyJ);
1172         tmp.multiplyByScalar(-1.0);
1173         
1174         result.add(tmp);
1175         
1176         //-1 * (c1 + x*c2) * (d1 + x*d2)^2 * (j1 + x*j2)
1177         tmp = polyC.multiplyAndReturnNew(polyD);
1178         tmp.multiply(polyD);
1179         tmp.multiply(polyJ);
1180         tmp.multiplyByScalar(-1.0);
1181         
1182         result.add(tmp);
1183         
1184         //-2 * (d1 + x*d2) * (f1 + x*f2) * (h1 + x*h2) * (i1 + x*i2)
1185         tmp = polyD.multiplyAndReturnNew(polyF);
1186         tmp.multiply(polyH);
1187         tmp.multiply(polyI);
1188         tmp.multiplyByScalar(-2.0);
1189         
1190         result.add(tmp);
1191         
1192         //-2 * (d1 + x*d2) * (e1 + x*e2) * (g1 + x*g2) * (i1 + x*i2)
1193         tmp = polyD.multiplyAndReturnNew(polyE);
1194         tmp.multiply(polyG);
1195         tmp.multiply(polyI);
1196         tmp.multiplyByScalar(-2.0);
1197         
1198         result.add(tmp);
1199         
1200         //2 * (c1 + x*c2) * (d1 + x*d2) * (g1 + x*g2) * (h1 + x*h2)
1201         tmp = polyC.multiplyAndReturnNew(polyD);
1202         tmp.multiply(polyG);
1203         tmp.multiply(polyH);
1204         tmp.multiplyByScalar(2.0);
1205         
1206         result.add(tmp);
1207         
1208         //(d1 + x*d2)^2 * (i1 + x*i2)^2
1209         tmp = polyD.multiplyAndReturnNew(polyD);
1210         tmp.multiply(polyI);
1211         tmp.multiply(polyI);
1212
