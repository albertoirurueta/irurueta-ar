79      /**
80       * Provides a new accelerometer sample to update SLAM estimation.
81       * This method must be called whenever the accelerometer sensor receives new
82       * data.
83       * If reconstructor is not running, calling this method has no effect.
84       * @param timestamp timestamp of accelerometer sample since epoch time and
85       * expressed in nanoseconds.
86       * @param accelerationX linear acceleration along x-axis expressed in meters
87       * per squared second (m/s^2).
88       * @param accelerationY linear acceleration along y-axis expressed in meters
89       * per squared second (m/s^2).
90       * @param accelerationZ linear acceleration along z-axis expressed in meters
91       * per squared second (m/s^2).
92       */
93      public void updateAccelerometerSample(long timestamp, float accelerationX,
94                                            float accelerationY, float accelerationZ) {
95          if (mLastViewPairTimestamp < 0) {
96              mLastViewPairTimestamp = timestamp;
97          }
98  
99          if (mSlamEstimator != null) {
100             mSlamEstimator.updateAccelerometerSample(timestamp - mLastViewPairTimestamp,
101                     accelerationX, accelerationY, accelerationZ);
102         }
103         mLastTimestamp = timestamp;
104     }

234     /**
235      * Transforms metric cameras on current pair of views so that they are referred to
236      * last kept location and rotation and upgrades cameras from metric stratum to
237      * euclidean stratum.
238      * @param isInitialPairOfViews true if initial pair of views is being processed, false otherwise.
239      * @param hasAbsoluteOrientation true if absolute orientation is required, false otherwise.
240      * @return true if cameras were successfully transformed.
241      */
242     @Override
243     protected boolean transformPairOfCamerasAndPoints(boolean isInitialPairOfViews,
244                                                       boolean hasAbsoluteOrientation) {
245         PinholeCamera previousMetricCamera = mPreviousMetricEstimatedCamera.getCamera();
246         PinholeCamera currentMetricCamera = mCurrentMetricEstimatedCamera.getCamera();
247         if (previousMetricCamera == null || currentMetricCamera == null) {
248             return false;
249         }
250 
251         mCurrentScale = estimateCurrentScale();
252         double sqrScale = mCurrentScale * mCurrentScale;
253 
254         mReferenceEuclideanTransformation = new MetricTransformation3D(mCurrentScale);
255         if (hasAbsoluteOrientation) {
256             mInvEuclideanCameraRotation = mLastEuclideanCameraRotation.inverseRotationAndReturnNew();
257             if (isInitialPairOfViews) {
258                 mReferenceEuclideanTransformation.setRotation(mInvEuclideanCameraRotation);
259             }
260         }
261 
262         if (!isInitialPairOfViews) {
263             //additional pairs also need to translate and rotate
264             if (mInvEuclideanCameraRotation == null) {
265                 mInvEuclideanCameraRotation = mLastEuclideanCameraRotation.inverseRotationAndReturnNew();
266             } else {
267                 mLastEuclideanCameraRotation.inverseRotation(mInvEuclideanCameraRotation);
268             }
269             mReferenceEuclideanTransformation.setRotation(mInvEuclideanCameraRotation);
270             mReferenceEuclideanTransformation.setTranslation(mLastEuclideanCameraCenter);
271         }
272 
273         try {
274             //transform cameras
275             PinholeCamera previousEuclideanCamera = mReferenceEuclideanTransformation.
276                     transformAndReturnNew(previousMetricCamera);
277             PinholeCamera currentEuclideanCamera = mReferenceEuclideanTransformation.
278                     transformAndReturnNew(currentMetricCamera);
279 
280             mPreviousEuclideanEstimatedCamera = new EstimatedCamera();
281             mPreviousEuclideanEstimatedCamera.setCamera(previousEuclideanCamera);
282             mPreviousEuclideanEstimatedCamera.setViewId(
283                     mPreviousMetricEstimatedCamera.getViewId());
284             mPreviousEuclideanEstimatedCamera.setQualityScore(
285                     mPreviousMetricEstimatedCamera.getQualityScore());
286             if (mPreviousMetricEstimatedCamera.getCovariance() != null) {
287                 mPreviousEuclideanEstimatedCamera.setCovariance(
288                         mPreviousMetricEstimatedCamera.getCovariance().
289                                 multiplyByScalarAndReturnNew(sqrScale));
290             }
291 
292             mCurrentEuclideanEstimatedCamera = new EstimatedCamera();
293             mCurrentEuclideanEstimatedCamera.setCamera(currentEuclideanCamera);
294             mCurrentEuclideanEstimatedCamera.setViewId(
295                     mCurrentMetricEstimatedCamera.getViewId());
296             mCurrentEuclideanEstimatedCamera.setQualityScore(
297                     mCurrentMetricEstimatedCamera.getQualityScore());
298             if (mCurrentMetricEstimatedCamera.getCovariance() != null) {
299                 mCurrentEuclideanEstimatedCamera.setCovariance(
300                         mCurrentMetricEstimatedCamera.getCovariance().
301                                 multiplyByScalarAndReturnNew(sqrScale));
302             }
303 
304             //transform points
305             mEuclideanReconstructedPoints = new ArrayList<>();
306             ReconstructedPoint3D euclideanReconstructedPoint;
307             Point3D metricPoint;
308             Point3D euclideanPoint;
309             for (ReconstructedPoint3D metricReconstructedPoint : mMetricReconstructedPoints) {
310                 metricPoint = metricReconstructedPoint.getPoint();
311                 euclideanPoint = mReferenceEuclideanTransformation.transformAndReturnNew(
312                         metricPoint);
313                 euclideanReconstructedPoint = new ReconstructedPoint3D();
314                 euclideanReconstructedPoint.setPoint(euclideanPoint);
315                 euclideanReconstructedPoint.setInlier(metricReconstructedPoint.isInlier());
316                 euclideanReconstructedPoint.setId(metricReconstructedPoint.getId());
317                 euclideanReconstructedPoint.setColorData(
318                         metricReconstructedPoint.getColorData());
319                 if (metricReconstructedPoint.getCovariance() != null) {
320                     euclideanReconstructedPoint.setCovariance(
321                             metricReconstructedPoint.getCovariance().
322                             multiplyByScalarAndReturnNew(sqrScale));
323                 }
324                 euclideanReconstructedPoint.setQualityScore(
325                         metricReconstructedPoint.getQualityScore());
326                 mEuclideanReconstructedPoints.add(euclideanReconstructedPoint);
327             }
328 
329         } catch (AlgebraException e) {
330             return false;
331         }
332 
333         return super.transformPairOfCamerasAndPoints(isInitialPairOfViews, hasAbsoluteOrientation);
334     }

336     /**
337      * Estimates current scale using SLAM data.
338      * @return estimated scale.
339      */
340     private double estimateCurrentScale() {
341         try {
342             PinholeCamera metricCamera1 = mPreviousMetricEstimatedCamera.getCamera();
343             PinholeCamera metricCamera2 = mCurrentMetricEstimatedCamera.getCamera();
344 
345             if (!metricCamera1.isCameraCenterAvailable()) {
346                 metricCamera1.decompose(false, true);
347             }
348             if (!metricCamera2.isCameraCenterAvailable()) {
349                 metricCamera2.decompose(false, true);
350             }
351 
352             Point3D metricCenter1 = metricCamera1.getCameraCenter();
353             Point3D metricCenter2 = metricCamera2.getCameraCenter();
354 
355             //obtain baseline (camera separation from slam estimator data)
356             double slamPosX = mSlamEstimator.getStatePositionX();
357             double slamPosY = mSlamEstimator.getStatePositionY();
358             double slamPosZ = mSlamEstimator.getStatePositionZ();
359 
360             mSlamPosition.setInhomogeneousCoordinates(slamPosX, slamPosY, slamPosZ);
361 
362             //we assume that euclidean center of 1st camera is at origin because by the time
363             //this method is called, metric cameras have not yet been orientation and position
364             // transformed
365             double euclideanBaseline = Math.sqrt(
366                     slamPosX * slamPosX +
367                     slamPosY * slamPosY +
368                     slamPosZ * slamPosZ);
369             double metricBaseline = metricCenter1.distanceTo(metricCenter2);
370 
371             //because when we call reset in SLAM estimator, timestamp is lost, we keep
372             //track of last timestamp to be subtracted on subsequent update calls
373             mLastViewPairTimestamp = mLastTimestamp;
374 
375             //reset linear velocity and orientation and keep other slam state parameters
376             mSlamEstimator.resetPositionAndVelocity();
377 
378             return euclideanBaseline / metricBaseline;
379         } catch (Exception e) {
380             mFailed = true;
381             //noinspection all
382             mListener.onFail((R)this);
383             return DEFAULT_SCALE;
384         }
385     }

387     /**
388      * Notifies SLAM state if notification is enabled at configuration time.
389      */
390     private void notifySlamStateIfNeeded() {
391         if (!mConfiguration.isNotifyAvailableSlamDataEnabled()) {
392             return;
393         }
394 
395         double lastPosX = mLastEuclideanCameraCenter != null ?
396                 mLastEuclideanCameraCenter.getInhomX() : 0.0;
397         double lastPosY = mLastEuclideanCameraCenter != null ?
398                 mLastEuclideanCameraCenter.getInhomY() : 0.0;
399         double lastPosZ = mLastEuclideanCameraCenter != null ?
400                 mLastEuclideanCameraCenter.getInhomZ() : 0.0;
401 
402         double positionX = lastPosX + mSlamEstimator.getStatePositionX();
403         double positionY = lastPosY + mSlamEstimator.getStatePositionY();
404         double positionZ = lastPosZ + mSlamEstimator.getStatePositionZ();
405 
406         double velocityX = mSlamEstimator.getStateVelocityX();
407         double velocityY = mSlamEstimator.getStateVelocityY();
408         double velocityZ = mSlamEstimator.getStateVelocityZ();
409 
410         double accelerationX = mSlamEstimator.getStateAccelerationX();
411         double accelerationY = mSlamEstimator.getStateAccelerationY();
412         double accelerationZ = mSlamEstimator.getStateAccelerationZ();
413 
414         double quaternionA = mSlamEstimator.getStateQuaternionA();
415         double quaternionB = mSlamEstimator.getStateQuaternionB();
416         double quaternionC = mSlamEstimator.getStateQuaternionC();
417         double quaternionD = mSlamEstimator.getStateQuaternionD();
418 
419         double angularSpeedX = mSlamEstimator.getStateAngularSpeedX();
420         double angularSpeedY = mSlamEstimator.getStateAngularSpeedY();
421         double angularSpeedZ = mSlamEstimator.getStateAngularSpeedZ();
422 
423         //noinspection all
424         mListener.onSlamDataAvailable((R)this, positionX, positionY, positionZ,
425                 velocityX, velocityY, velocityZ,
426                 accelerationX, accelerationY, accelerationZ,
427                 quaternionA, quaternionB, quaternionC, quaternionD,
428                 angularSpeedX, angularSpeedY, angularSpeedZ, mSlamEstimator.getStateCovariance());
429     }

431     /**
432      * Notifies estimated camera by means of SLAM if notification is enabled at
433      * configuration time and intrinsics are already available.
434      */
435     private void notifySlamCameraIfNeeded() {
436         if (!mConfiguration.isNotifyEstimatedSlamCameraEnabled()) {
437             return;
438         }
439 
440         //try with current camera
441         PinholeCamera camera = mCurrentEuclideanEstimatedCamera != null ?
442                 mCurrentEuclideanEstimatedCamera.getCamera() : null;
443         if (camera == null) {
444             //if not available try with previous camera
445             camera = mPreviousEuclideanEstimatedCamera != null ?
446                     mPreviousEuclideanEstimatedCamera.getCamera() : null;
447         }
448 
449         try {
450             PinholeCameraIntrinsicParameters intrinsicParameters = null;
451             if (camera != null) {
452                 if (!camera.areIntrinsicParametersAvailable()) {
453                     //decompose camera to obtain intrinsic parameters
454                     camera.decompose();
455                 }
456 
457                 intrinsicParameters = camera.getIntrinsicParameters();
458             } else if (mConfiguration.areIntrinsicParametersKnown()) {
459                 //noinspection unchecked
460                 intrinsicParameters = mListener.onIntrinsicParametersRequested((R)this, mCurrentViewId);
461             }
462 
463             if (intrinsicParameters == null) {
464                 return;
465             }
466 
467             double lastPosX = mLastEuclideanCameraCenter != null ?
468                     mLastEuclideanCameraCenter.getInhomX() : 0.0;
469             double lastPosY = mLastEuclideanCameraCenter != null ?
470                     mLastEuclideanCameraCenter.getInhomY() : 0.0;
471             double lastPosZ = mLastEuclideanCameraCenter != null ?
472                     mLastEuclideanCameraCenter.getInhomZ() : 0.0;
473 
474             double positionX = lastPosX + mSlamEstimator.getStatePositionX();
475             double positionY = lastPosY + mSlamEstimator.getStatePositionY();
476             double positionZ = lastPosZ + mSlamEstimator.getStatePositionZ();
477             mSlamPosition.setInhomogeneousCoordinates(positionX, positionY, positionZ);
478 
479             double quaternionA = mSlamEstimator.getStateQuaternionA();
480             double quaternionB = mSlamEstimator.getStateQuaternionB();
481             double quaternionC = mSlamEstimator.getStateQuaternionC();
482             double quaternionD = mSlamEstimator.getStateQuaternionD();
483             mSlamRotation.setA(quaternionA);
484             mSlamRotation.setB(quaternionB);
485             mSlamRotation.setC(quaternionC);
486             mSlamRotation.setD(quaternionD);
487 
488             mSlamCamera.setIntrinsicAndExtrinsicParameters(intrinsicParameters, mSlamRotation,
489                     mSlamPosition);
490 
491             //noinspection all
492             mListener.onSlamCameraEstimated((R)this, mSlamCamera);
493 
494         } catch (GeometryException ignore) { /* do nothing */ }
495     }
