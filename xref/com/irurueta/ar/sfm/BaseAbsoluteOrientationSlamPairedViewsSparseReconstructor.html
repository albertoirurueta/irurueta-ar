17  package com.irurueta.ar.sfm;
18  
19  import com.irurueta.ar.slam.AbsoluteOrientationBaseSlamEstimator;
20  import com.irurueta.ar.slam.BaseCalibrationData;
21  import com.irurueta.geometry.Rotation3D;
22  
23  /**
24   * Base class in charge of estimating pairs of cameras and 3D reconstructed points from sparse
25   * image point correspondences in multiple view pairs and also in charge of estimating overall
26   * scene scale and absolute orientation by means of SLAM (Simultaneous Location And Mapping)
27   * using data obtained from sensors like accelerometers or gyroscopes.
28   * NOTE: absolute orientation slam estimators are not very accurate during estimation of
29   * the orientation state, for that reason we take into account the initial orientation.
30   *
31   * @param <D> calibrator type associated to implementations of SLAM calibration
32   *            data.
33   * @param <C> type of configuration.
34   * @param <R> type of re-constructor.
35   * @param <L> type of listener.
36   * @param <S> type of SLAM estimator.
37   */
38  public abstract class BaseAbsoluteOrientationSlamPairedViewsSparseReconstructor<
39          D extends BaseCalibrationData,
40          C extends BaseSlamPairedViewsSparseReconstructorConfiguration<D, C>,
41          R extends BaseSlamPairedViewsSparseReconstructor<D, C, R, L, S>,
42          L extends BaseSlamPairedViewsSparseReconstructorListener<R>,
43          S extends AbsoluteOrientationBaseSlamEstimator<D>> extends
44          BaseSlamPairedViewsSparseReconstructor<D, C, R, L, S> {
45  
46      /**
47       * First sample of orientation received.
48       */
49      private Rotation3D firstOrientation;
50  
51      /**
52       * Constructor.
53       *
54       * @param configuration configuration for this re-constructor.
55       * @param listener      listener in charge of handling events.
56       * @throws NullPointerException if listener or configuration is not
57       *                              provided.
58       */
59      protected BaseAbsoluteOrientationSlamPairedViewsSparseReconstructor(final C configuration, final L listener) {
60          super(configuration, listener);
61      }
62  
63      /**
64       * Provides a new orientation sample to update SLAM estimator.
65       * If re-constructor is not running, calling this method has no effect.
66       *
67       * @param timestamp   timestamp of accelerometer sample since epoch time and
68       *                    expressed in nanoseconds.
69       * @param orientation new orientation.
70       */
71      public void updateOrientationSample(final long timestamp, final Rotation3D orientation) {
72          if (slamEstimator != null) {
73              slamEstimator.updateOrientationSample(timestamp, orientation);
74          }
75          if (firstOrientation == null) {
76              // make a copy of orientation
77              firstOrientation = orientation.toQuaternion();
78          }
79      }
80  
81      /**
82       * Indicates whether implementations of a re-constructor uses absolute orientation or
83       * not.
84       *
85       * @return true if absolute orientation is used, false, otherwise.
86       */
87      @Override
88      protected boolean hasAbsoluteOrientation() {
89          return true;
90      }
91  
92      /**
93       * Transforms metric cameras on current pair of views so that they are referred to
94       * last kept location and rotation and upgrades cameras from metric stratum to
95       * Euclidean stratum.
96       *
97       * @param isInitialPairOfViews   true if initial pair of views is being processed, false otherwise.
98       * @param hasAbsoluteOrientation true if absolute orientation is required, false otherwise.
99       * @return true if cameras were successfully transformed.
100      */
101     @Override
102     protected boolean transformPairOfCamerasAndPoints(
103             final boolean isInitialPairOfViews, final boolean hasAbsoluteOrientation) {
104 
105         // R1' = R1*Rdiff
106         // Rdiff = R1^T*R1'
107 
108         // where R1' is the desired orientation (obtained by sampling a
109         // sensor)
110         // and R1 is always the identity for the 1st camera.
111         // Hence R1' = Rdiff
112 
113         // t1' is the desired translation which is zero for the 1st
114         // camera.
115 
116         // We want: P1' = K*[R1' t1'] = K*[R1' 0]
117         // And we have P1 = K[I 0]
118 
119         // We need a transformation T so that:
120         // P1' = P1*T^-1 = K[I 0][R1' 0]
121         //                       [0   1]
122 
123         // Hence: T^-1 = [R1' 0]
124         //               [0   1]
125 
126         // or T = [R1'^T 0]
127         //        [0     1]
128 
129         // because we are also applying a transformation of scale s,
130         // the combination of both transformations is
131         // T = [s*R1'^T 0]
132         //     [0       1]
133 
134         if (isInitialPairOfViews && hasAbsoluteOrientation) {
135             // The first pair of views does not require setting translation and rotation
136             lastEuclideanCameraRotation = firstOrientation;
137         }
138 
139         return super.transformPairOfCamerasAndPoints(isInitialPairOfViews, hasAbsoluteOrientation);
140     }
141 }
