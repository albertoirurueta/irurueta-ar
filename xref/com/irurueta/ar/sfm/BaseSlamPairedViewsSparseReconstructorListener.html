17  package com.irurueta.ar.sfm;
18  
19  import com.irurueta.algebra.Matrix;
20  import com.irurueta.ar.slam.AbsoluteOrientationConstantVelocityModelSlamEstimator;
21  import com.irurueta.ar.slam.AbsoluteOrientationSlamEstimator;
22  import com.irurueta.ar.slam.ConstantVelocityModelSlamEstimator;
23  import com.irurueta.ar.slam.SlamEstimator;
24  import com.irurueta.geometry.PinholeCamera;
25  
26  /**
27   * Listener to retrieve and store required data to compute a 3D reconstruction from sparse
28   * image point correspondences in multiple views.
29   * Implementations of this interface
30   *
31   * @param <R> re-constructor type.
32   */
33  public interface BaseSlamPairedViewsSparseReconstructorListener<R
34          extends BaseSlamPairedViewsSparseReconstructor<?, ?, ?, ?, ?>>
35          extends BasePairedViewsSparseReconstructorListener<R> {
36  
37      /**
38       * Called whenever slam data notification is enabled and each time all required samples (accelerometer,
39       * gyroscope or orientation) are received in order to update SLAM system state to notify
40       * new position, velocity, acceleration, orientation and angular speed.
41       *
42       * @param reconstructor re-constructor raising this event.
43       * @param positionX     x position coordinate expressed in meters (m).
44       * @param positionY     y position coordinate expressed in meters (m).
45       * @param positionZ     z position coordinate expressed in meters (m).
46       * @param velocityX     x velocity coordinate expressed in meters per second (m/s).
47       * @param velocityY     y velocity coordinate expressed in meters per second (m/s).
48       * @param velocityZ     z velocity coordinate expressed in meters per second (m/s).
49       * @param accelerationX x linear acceleration coordinate expressed in meters per squared second (m/s^2).
50       * @param accelerationY y linear acceleration coordinate expressed in meters per squared second (m/s^2).
51       * @param accelerationZ z linear acceleration coordinate expressed in meters per squared second (m/s^2).
52       * @param quaternionA   a component of quaternion expressing current orientation.
53       * @param quaternionB   b component of quaternion expressing current orientation.
54       * @param quaternionC   c component of quaternion expressing current orientation.
55       * @param quaternionD   d component of quaternion expressing current orientation.
56       * @param angularSpeedX x coordinate of angular speed expressed in radians per second (rad/s).
57       * @param angularSpeedY y coordinate of angular speed expressed in radians per second (rad/s).
58       * @param angularSpeedZ z coordinate of angular speed expressed in radians per second (rad/s).
59       * @param covariance    contains covariance matrix of estimated SLAM state. Matrix meaning will change
60       *                      depending on slam implementation. See: {@link SlamEstimator#getStateCovariance()},
61       *                      {@link ConstantVelocityModelSlamEstimator#getStateCovariance()},
62       *                      {@link AbsoluteOrientationSlamEstimator#getStateCovariance()} or
63       *                      {@link AbsoluteOrientationConstantVelocityModelSlamEstimator#getStateCovariance()}.
64       */
65      void onSlamDataAvailable(final R reconstructor, final double positionX, final double positionY,
66                               final double positionZ, final double velocityX, final double velocityY,
67                               final double velocityZ, final double accelerationX, final double accelerationY,
68                               final double accelerationZ, final double quaternionA, final double quaternionB,
69                               final double quaternionC, final double quaternionD, final double angularSpeedX,
70                               final double angularSpeedY, final double angularSpeedZ, final Matrix covariance);
71  
72      /**
73       * Called whenever estimated SLAM camera notification is enabled.
74       * This method is called each time all required samples (accelerometer, gyroscope or orientation) are
75       * received in order to update SLAM system state to notify a new camera containing
76       * current intrinsic parameters, position and orientation.
77       *
78       * @param reconstructor re-constructor raising this event.
79       * @param camera        current camera estimated using IMU data. This instance and its associated
80       *                      instances (camera center, rotation and intrinsic parameters) will be reused
81       *                      between consecutive calls to this method.
82       */
83      void onSlamCameraEstimated(final R reconstructor, final PinholeCamera camera);
84  }
