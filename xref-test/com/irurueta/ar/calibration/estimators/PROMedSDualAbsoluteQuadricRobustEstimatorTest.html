805     @Test
806     public void testEstimate() throws AlgebraException, 
807             InvalidTransformationException, LockedException, NotReadyException, 
808             CameraException, NotAvailableException {
809         
810         int numSucceeded = 0;
811         for (int t = 0; t < 10*TIMES; t++) {
812             UniformRandomizer randomizer = new UniformRandomizer(new Random());
813         
814             //create ground truth intrinsic parameters
815             double horizontalFocalLength, verticalFocalLength, skewness,
816                     horizontalPrincipalPoint, verticalPrincipalPoint;
817             double aspectRatio = 1.0;
818             horizontalFocalLength = randomizer.nextDouble(MIN_FOCAL_LENGTH, 
819                     MAX_FOCAL_LENGTH);
820             verticalFocalLength = aspectRatio * horizontalFocalLength;
821             skewness = horizontalPrincipalPoint = verticalPrincipalPoint = 0.0;
822         
823             PinholeCameraIntrinsicParameters metricIntrinsic =
824                     new PinholeCameraIntrinsicParameters(horizontalFocalLength,
825                     verticalFocalLength, horizontalPrincipalPoint,
826                     verticalPrincipalPoint, skewness);
827         
828             DualImageOfAbsoluteConic metricDiac = new DualImageOfAbsoluteConic(
829                     metricIntrinsic);
830             metricDiac.normalize();
831             Matrix metricDiacMatrix = metricDiac.asMatrix();
832         
833             //generate random projective transformation to transform ground 
834             //truth cameras
835             Matrix T = Matrix.createWithUniformRandomValues(
836                     ProjectiveTransformation3D.HOM_COORDS, 
837                     ProjectiveTransformation3D.HOM_COORDS, MIN_RANDOM_VALUE, 
838                     MAX_RANDOM_VALUE);
839         
840             //ensure last element is not zero
841             T.setElementAt(ProjectiveTransformation3D.HOM_COORDS - 1, 
842                     ProjectiveTransformation3D.HOM_COORDS - 1, 1.0);        
843 
844             ProjectiveTransformation3D transformation = 
845                     new ProjectiveTransformation3D(T);
846         
847             transformation.normalize();
848         
849             DualAbsoluteQuadric projectiveDaq = new DualAbsoluteQuadric(
850                     transformation);
851             projectiveDaq.normalize();
852                 
853             Matrix projectiveDaqMatrix = projectiveDaq.asMatrix();
854         
855             double roll, pitch, yaw, x, y, z;
856             Quaternion q;
857             InhomogeneousPoint3D cameraCenter;
858             PinholeCamera metricCamera, projectiveCamera;
859             List<PinholeCamera> metricCameras = new ArrayList<>();
860             List<PinholeCamera> projectiveCameras = 
861                     new ArrayList<>();
862         
863             PROMedSDualAbsoluteQuadricRobustEstimator estimator =
864                     new PROMedSDualAbsoluteQuadricRobustEstimator();
865             estimator.setListener(this);
866             estimator.setZeroSkewness(true);
867             estimator.setPrincipalPointAtOrigin(true);
868             estimator.setFocalDistanceAspectRatioKnown(true);
869             estimator.setFocalDistanceAspectRatio(1.0);
870             estimator.setSingularityEnforced(false);
871         
872             GaussianRandomizer errorRandomizer = new GaussianRandomizer(
873                     new Random(), 0.0, STD_ERROR);
874             boolean[] inliers = new boolean[NUM_CAMS];
875             double[] qualityScores = new double[NUM_CAMS];
876             for (int i = 0; i < NUM_CAMS; i++) {
877                 roll = randomizer.nextDouble(2.0 * MIN_ANGLE_DEGREES, 
878                         2.0 * MAX_ANGLE_DEGREES) * Math.PI / 180.0;
879                 pitch = randomizer.nextDouble(MIN_ANGLE_DEGREES, 
880                         MAX_ANGLE_DEGREES) * Math.PI / 180.0;
881                 yaw = randomizer.nextDouble(2.0 * MIN_ANGLE_DEGREES, 
882                         2.0 * MAX_ANGLE_DEGREES) * Math.PI / 180.0;
883 
884                 q = new Quaternion(roll, pitch, yaw);
885             
886                 x = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
887                 y = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
888                 z = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
889                 cameraCenter = new InhomogeneousPoint3D(x, y, z);
890             
891                 if (randomizer.nextInt(0, 100) < PERCENTAGE_OUTLIERS) {
892                     //outlier (add error to metric intrinsics)
893                     inliers[i] = false;
894                     double errorHorizontalFocalLength = 
895                             errorRandomizer.nextDouble();
896                     double errorAspectRatio = errorRandomizer.nextDouble();
897                     double errorSkewness = errorRandomizer.nextDouble();
898                     double errorHorizontalPrincipalPoint = 
899                             errorRandomizer.nextDouble();
900                     double errorVerticalPrincipalPoint =
901                             errorRandomizer.nextDouble();
902                     
903                     double avgAbsError = (Math.abs(errorHorizontalFocalLength) + 
904                             Math.abs(errorAspectRatio) + 
905                             Math.abs(errorSkewness) +
906                             Math.abs(errorHorizontalPrincipalPoint) +
907                             Math.abs(errorVerticalPrincipalPoint)) / 5.0;
908                     qualityScores[i] = 1.0 / (1.0 + avgAbsError);
909                 
910                     double outlierHorizontalFocalLength = 
911                             horizontalFocalLength + errorHorizontalFocalLength;
912                     double outlierAspectRatio = aspectRatio + errorAspectRatio;
913                     double outlierVerticalFocalLength = 
914                             outlierAspectRatio * outlierHorizontalFocalLength;
915                     double outlierSkewness = skewness + errorSkewness;
916                     double outlierHorizontalPrincipalPoint = 
917                             horizontalPrincipalPoint + 
918                             errorHorizontalPrincipalPoint;
919                     double outlierVerticalPrincipalPoint =
920                             verticalPrincipalPoint + errorVerticalPrincipalPoint;
921                 
922                     PinholeCameraIntrinsicParameters outlierMetricIntrinsic = 
923                             new PinholeCameraIntrinsicParameters(
924                             outlierHorizontalFocalLength,
925                             outlierVerticalFocalLength, 
926                             outlierHorizontalPrincipalPoint,
927                             outlierVerticalPrincipalPoint, outlierSkewness);
928                       
929                     metricCamera = new PinholeCamera(outlierMetricIntrinsic, q, 
930                             cameraCenter);
931                 } else {
932                     //inlier
933                     inliers[i] = true;
934                     qualityScores[i] = 1.0;
935                     metricCamera = new PinholeCamera(metricIntrinsic, q, 
936                         cameraCenter);
937                 }
938                 metricCamera.normalize();   
939                 metricCameras.add(metricCamera);
940             
941                 //transform camera
942                 projectiveCamera = transformation.transformAndReturnNew(
943                         metricCamera);
944             
945                 projectiveCameras.add(projectiveCamera);            
946             }
947         
948             estimator.setCameras(projectiveCameras);
949             estimator.setQualityScores(qualityScores);
950         
951             try {
952                 DualAbsoluteQuadric estimatedDaq = estimator.estimate();
953                 estimatedDaq.normalize();
954                 Matrix estimatedDaqMatrix = estimatedDaq.asMatrix();
955 
956                 //check that DAQ has rank 3 (zero determinant)
957                 if (Math.abs(Utils.det(estimatedDaqMatrix)) > ABSOLUTE_ERROR) {
958                     continue;
959                 }
960                 assertEquals(Utils.det(estimatedDaqMatrix), 0.0, 
961                         ABSOLUTE_ERROR);
962                 
963                 if (!projectiveDaqMatrix.equals(estimatedDaqMatrix,
964                         ABSOLUTE_ERROR)) {
965                     continue;
966                 }
967                 assertTrue(projectiveDaqMatrix.equals(estimatedDaqMatrix, 
968                         ABSOLUTE_ERROR));
969                 
970                 ProjectiveTransformation3D estimatedTransformation =
971                         estimatedDaq.getMetricToProjectiveTransformation();
972                 estimatedTransformation.normalize();
973                 ProjectiveTransformation3D invEstimatedTransformation =
974                         (ProjectiveTransformation3D)estimatedTransformation.
975                                 inverseAndReturnNew();
976                 
977                 //projected estimated DAQ using projective cameras to obtain 
978                 //DIAC and check that DIAC in projective stratum is equal to 
979                 //DIAC in metric stratum (for inlier cameras only)
980                 Point3D previousEstimatedMetricCenter = null;
981                 Point3D previousMetricCenter = null;
982                 Point3D estimatedMetricCenter, metricCenter;
983                 Rotation3D previousEstimatedMetricRotation = null;
984                 Rotation3D previousMetricRotation = null;
985                 Rotation3D estimatedMetricRotation, metricRotation;
986                 double distanceEstimatedCenter, distanceCenter;
987                 double previousScale = 1.0, scale = 1.0;                                
988                 PinholeCamera estimatedMetricCamera;
989                 boolean anyFailed = false;
990                 int count = 0;                
991                 for (int i = 0; i < NUM_CAMS; i++) {
992                     if (!inliers[i]) {
993                         continue;
994                     }
995                     
996                     projectiveCamera = projectiveCameras.get(i);
997                     
998                     DualImageOfAbsoluteConic projectedProjectiveDiac =
999                             new DualImageOfAbsoluteConic(projectiveCamera,
1000                             estimatedDaq);
1001                     projectedProjectiveDiac.normalize();
1002                     
1003                     Matrix projectedProjectiveDiacMatrix =
1004                             projectedProjectiveDiac.asMatrix();
1005                     
1006                     if (!metricDiacMatrix.equals(projectedProjectiveDiacMatrix,
1007                             ABSOLUTE_ERROR)) {
1008                         anyFailed = true;
1009                         continue;
1010                     }
1011                     assertTrue(metricDiacMatrix.equals(
1012                             projectedProjectiveDiacMatrix, ABSOLUTE_ERROR));
1013                     
1014                     estimatedMetricCamera = invEstimatedTransformation.
1015                             transformAndReturnNew(projectiveCamera);
1016                     
1017                     estimatedMetricCamera.decompose();
1018                     PinholeCameraIntrinsicParameters estimatedIntrinsic =
1019                             estimatedMetricCamera.getIntrinsicParameters();
1020                     
1021                     assertEquals(horizontalFocalLength,
1022                             estimatedIntrinsic.getHorizontalFocalLength(),
1023                             5*LARGE_ABSOLUTE_ERROR);
1024                     assertEquals(verticalFocalLength,
1025                             estimatedIntrinsic.getVerticalFocalLength(),
1026                             5*LARGE_ABSOLUTE_ERROR);
1027                     assertEquals(skewness, estimatedIntrinsic.getSkewness(),
1028                             5*LARGE_ABSOLUTE_ERROR);
1029                     assertEquals(horizontalPrincipalPoint,
1030                             estimatedIntrinsic.getHorizontalPrincipalPoint(),
1031                             5*LARGE_ABSOLUTE_ERROR);
1032                     assertEquals(verticalPrincipalPoint,
1033                             estimatedIntrinsic.getVerticalPrincipalPoint(),
1034                             5*LARGE_ABSOLUTE_ERROR);
1035                     
1036                     if (anyFailed) {
1037                         continue;
1038                     }
1039                     
1040                     //check that when DAQ is successfully estimated, estimated
1041                     //metric cameras are in the metric stratum up to an 
1042                     //arbitrary scale
1043                     
1044                     metricCamera = metricCameras.get(i);
1045                     metricCamera.decompose();
1046                     
1047                     estimatedMetricCenter = estimatedMetricCamera.
1048                             getCameraCenter();
1049                     metricCenter = metricCamera.getCameraCenter();
1050                     estimatedMetricRotation = estimatedMetricCamera.
1051                             getCameraRotation();
1052                     metricRotation = metricCamera.getCameraRotation();
1053                                         
1054                     if (count > 0) {
1055                         distanceEstimatedCenter = previousEstimatedMetricCenter.
1056                                 distanceTo(estimatedMetricCenter);
1057                         distanceCenter = previousMetricCenter.distanceTo(
1058                                 metricCenter);
1059                         scale = distanceEstimatedCenter / distanceCenter;
1060                         
1061                         Rotation3D diffEstimatedRotation =
1062                                 estimatedMetricRotation.combineAndReturnNew(
1063                                 previousEstimatedMetricRotation.
1064                                 inverseRotationAndReturnNew());
1065                         Rotation3D diffRotation = metricRotation.
1066                                 combineAndReturnNew(previousMetricRotation.
1067                                 inverseRotationAndReturnNew());
1068                         
1069                         Matrix rot1 = diffEstimatedRotation.
1070                                 asInhomogeneousMatrix();
1071                         Matrix rot2 = diffRotation.asInhomogeneousMatrix();
1072                         assertTrue(rot1.equals(rot2, LARGE_ABSOLUTE_ERROR));
1073                     }
1074                     
1075                     if (count > 1) {
1076                         assertEquals(scale, previousScale, 5*LARGE_ABSOLUTE_ERROR);                    
1077                     }
1078                                         
1079                     previousEstimatedMetricCenter = estimatedMetricCenter;
1080                     previousMetricCenter = metricCenter;
1081                     previousScale = scale;         
1082                     
1083                     previousEstimatedMetricRotation = estimatedMetricRotation;
1084                     previousMetricRotation = metricRotation;                    
1085                     
1086                     count++;                    
1087                 }
1088                 
1089                 numSucceeded++;
1090                 
1091                 if (numSucceeded > 0) {
1092                     break;
1093                 }
1094             }  catch (RobustEstimatorException ignore) { }
1095         }
1096         
1097         //sometimes if cameras are in degenerate configurations, DAQ estimation
1098         //can fail, for that reason we check that algorithm at least works one
1099         //if we retry multiple times.
1100         assertTrue(numSucceeded > 0);
1101     }
