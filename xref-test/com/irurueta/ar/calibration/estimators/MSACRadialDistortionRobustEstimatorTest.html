561     @Test
562     void testEstimate() throws NotSupportedException, LockedException, NotReadyException, RobustEstimatorException,
563             DistortionException {
564         final var randomizer = new UniformRandomizer();
565 
566         for (var j = 0; j < TIMES; j++) {
567             final var k1 = randomizer.nextDouble(MIN_PARAM_VALUE, MAX_PARAM_VALUE);
568             final var k2 = randomizer.nextDouble(MIN_PARAM_VALUE, MAX_PARAM_VALUE);
569 
570             final var center = new InhomogeneousPoint2D(randomizer.nextDouble(MIN_POINT_VALUE, MAX_POINT_VALUE),
571                     randomizer.nextDouble(MIN_POINT_VALUE, MAX_POINT_VALUE));
572 
573             final var distortion = new RadialDistortion(k1, k2, center);
574 
575             final var nPoints = randomizer.nextInt(MIN_NUM_POINTS, MAX_NUM_POINTS);
576 
577             final var errorRandomizer = new GaussianRandomizer(0.0, STD_ERROR);
578             final var distortedPointsWithError = new ArrayList<Point2D>();
579             final var distortedPoints = new ArrayList<Point2D>();
580             final var undistortedPoints = new ArrayList<Point2D>();
581             Point2D distortedPoint;
582             Point2D distortedPointWithError;
583             Point2D undistortedPoint;
584             for (var i = 0; i < nPoints; i++) {
585                 undistortedPoint = new InhomogeneousPoint2D(randomizer.nextDouble(MIN_POINT_VALUE, MAX_POINT_VALUE),
586                         randomizer.nextDouble(MIN_POINT_VALUE, MAX_POINT_VALUE));
587 
588                 distortedPoint = distortion.distort(undistortedPoint);
589                 if (randomizer.nextInt(0, 100) < PERCENTAGE_OUTLIER) {
590                     // point is outlier
591                     final var errorX = errorRandomizer.nextDouble();
592                     final var errorY = errorRandomizer.nextDouble();
593                     distortedPointWithError = new InhomogeneousPoint2D(distortedPoint.getInhomX() + errorX,
594                             distortedPoint.getInhomY() + errorY);
595                 } else {
596                     // point is inlier
597                     distortedPointWithError = distortedPoint;
598                 }
599                 distortedPoints.add(distortedPoint);
600                 distortedPointsWithError.add(distortedPointWithError);
601                 undistortedPoints.add(undistortedPoint);
602             }
603 
604             final var estimator = new MSACRadialDistortionRobustEstimator(distortedPointsWithError, undistortedPoints,
605                     this);
606 
607             estimator.setIntrinsic(distortion.getIntrinsic());
608             estimator.setThreshold(THRESHOLD);
609 
610             assertEquals(0, estimateStart);
611             assertEquals(0, estimateEnd);
612             assertEquals(0, estimateNextIteration);
613             assertEquals(0, estimateProgressChange);
614             assertTrue(estimator.isReady());
615             assertFalse(estimator.isLocked());
616 
617             final var distortion2 = estimator.estimate();
618 
619             assertEquals(1, estimateStart);
620             assertEquals(1, estimateEnd);
621             assertTrue(estimateNextIteration > 0);
622             assertTrue(estimateProgressChange >= 0);
623             reset();
624 
625             // check correctness of estimation
626             assertEquals(k1, distortion2.getK1(), ABSOLUTE_ERROR);
627             assertEquals(k2, distortion2.getK2(), ABSOLUTE_ERROR);
628             assertEquals(distortion2.getCenter(), center);
629 
630             for (var i = 0; i < nPoints; i++) {
631                 assertEquals(distortedPoints.get(i), distortion2.distort(undistortedPoints.get(i)));
632             }
633         }
634     }
