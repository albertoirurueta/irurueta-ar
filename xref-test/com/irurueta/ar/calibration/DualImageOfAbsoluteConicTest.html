39      private static final double MAX_SKEWNESS = 1.0;

42      private static final double MAX_PRINCIPAL_POINT = 100.0;
43  
44      private static final double MIN_ANGLE_DEGREES = -90.0;
45      private static final double MAX_ANGLE_DEGREES = 90.0;
46      
47      private static final double ABSOLUTE_ERROR = 1e-8;
48      
49      private static final int DIAC_ROWS = 3;
50      private static final int DIAC_COLS = 3;

66      @Test
67      public void testConstructor() 
68              throws InvalidPinholeCameraIntrinsicParametersException, 
69              WrongSizeException,
70              NonSymmetricMatrixException {
71          //create intrinsic parameters
72          UniformRandomizer randomizer = new UniformRandomizer(new Random());
73          double horizontalFocalLength = randomizer.nextDouble(MIN_FOCAL_LENGTH, 
74                  MAX_FOCAL_LENGTH);
75          double verticalFocalLength = randomizer.nextDouble(MIN_FOCAL_LENGTH, 
76                  MAX_FOCAL_LENGTH);
77          
78          double skewness = randomizer.nextDouble(MIN_SKEWNESS, MAX_SKEWNESS);
79          
80          double horizontalPrincipalPoint = randomizer.nextDouble(
81                  MIN_PRINCIPAL_POINT, MAX_PRINCIPAL_POINT);
82          double verticalPrincipalPoint = randomizer.nextDouble(
83                  MIN_PRINCIPAL_POINT, MAX_PRINCIPAL_POINT);
84          //rotation
85          double alphaEuler = randomizer.nextDouble(
86                  MIN_ANGLE_DEGREES * Math.PI / 180.0, 
87                  MAX_ANGLE_DEGREES * Math.PI / 180.0);
88          double betaEuler = randomizer.nextDouble(
89                  MIN_ANGLE_DEGREES * Math.PI / 180.0, 
90                  MAX_ANGLE_DEGREES * Math.PI / 180.0);
91          double gammaEuler = randomizer.nextDouble(
92                  MIN_ANGLE_DEGREES * Math.PI / 180.0, 
93                  MAX_ANGLE_DEGREES * Math.PI / 180.0);
94          
95          MatrixRotation3D rotation = new MatrixRotation3D(alphaEuler, betaEuler, 
96                  gammaEuler);
97          //camera center
98          double[] cameraCenterArray = new double[INHOM_3D_COORDS];
99          randomizer.fill(cameraCenterArray, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
100         InhomogeneousPoint3D cameraCenter = new InhomogeneousPoint3D(
101                 cameraCenterArray);
102                 
103         PinholeCameraIntrinsicParameters intrinsic = 
104                 new PinholeCameraIntrinsicParameters(horizontalFocalLength,
105                 verticalFocalLength, horizontalPrincipalPoint, 
106                 verticalPrincipalPoint, skewness);
107         
108         PinholeCamera camera = new PinholeCamera(intrinsic, rotation, 
109                 cameraCenter);
110         
111         DualQuadric daq = DualQuadric.createCanonicalDualAbsoluteQuadric();
112         
113         //test constructor from instrinsic parameters
114         DualImageOfAbsoluteConic diac = new DualImageOfAbsoluteConic(intrinsic);
115         DualImageOfAbsoluteConic diac2 = new DualImageOfAbsoluteConic(camera, 
116                 daq);
117         
118         diac.normalize();
119         diac2.normalize();
120         
121         assertEquals(diac.getA(), diac2.getA(), 10.0*ABSOLUTE_ERROR);
122         assertEquals(diac.getB(), diac2.getB(), 10.0*ABSOLUTE_ERROR);
123         assertEquals(diac.getC(), diac2.getC(), 10.0*ABSOLUTE_ERROR);
124         assertEquals(diac.getD(), diac2.getD(), 10.0*ABSOLUTE_ERROR);
125         assertEquals(diac.getE(), diac2.getE(), 10.0*ABSOLUTE_ERROR);
126         assertEquals(diac.getF(), diac2.getF(), 10.0*ABSOLUTE_ERROR);
127         
128         PinholeCameraIntrinsicParameters intrinsic2 = 
129                 diac.getIntrinsicParameters();
130         
131         assertEquals(intrinsic.getHorizontalFocalLength(), 
132                 intrinsic2.getHorizontalFocalLength(), ABSOLUTE_ERROR);
133         assertEquals(intrinsic.getVerticalFocalLength(),
134                 intrinsic2.getVerticalFocalLength(), ABSOLUTE_ERROR);
135         assertEquals(intrinsic.getHorizontalPrincipalPoint(),
136                 intrinsic2.getHorizontalPrincipalPoint(), ABSOLUTE_ERROR);
137         assertEquals(intrinsic.getVerticalPrincipalPoint(),
138                 intrinsic2.getVerticalPrincipalPoint(), ABSOLUTE_ERROR);
139         assertEquals(intrinsic.getSkewness(), intrinsic2.getSkewness(),
140                 ABSOLUTE_ERROR);
141 
142         //test constructor from parameters
143         double a = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
144         double b = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
145         double c = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
146         double d = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
147         double e = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
148         double f = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
149 
150         diac = new DualImageOfAbsoluteConic(a, b, c, d, e, f);
151         
152         //check correctness
153         assertEquals(diac.getA(), a, 0.0);
154         assertEquals(diac.getB(), b, 0.0);
155         assertEquals(diac.getC(), c, 0.0);
156         assertEquals(diac.getD(), d, 0.0);
157         assertEquals(diac.getE(), e, 0.0);
158         assertEquals(diac.getF(), f, 0.0);
159         
160         //test constructor from matrix
161         Matrix m = new Matrix(DIAC_ROWS, DIAC_COLS);
162         //get random values
163         a = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
164         b = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
165         c = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
166         d = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
167         e = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
168         f = randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
169         
170         m.setElementAt(0, 0, a);
171         m.setElementAt(0, 1, b);
172         m.setElementAt(0, 2, d);
173         m.setElementAt(1, 0, b);
174         m.setElementAt(1, 1, c);
175         m.setElementAt(1, 2, e);
176         m.setElementAt(2, 0, d);
177         m.setElementAt(2, 1, e);
178         m.setElementAt(2, 2, f);
179         
180         diac = new DualImageOfAbsoluteConic(m);
181         
182         assertEquals(diac.getA(), m.getElementAt(0, 0), 0.0);
183         assertEquals(diac.getB(), m.getElementAt(0, 1), 0.0);
184         assertEquals(diac.getC(), m.getElementAt(1, 1), 0.0);
185         assertEquals(diac.getD(), m.getElementAt(0, 2), 0.0);
186         assertEquals(diac.getE(), m.getElementAt(1, 2), 0.0);
187         assertEquals(diac.getF(), m.getElementAt(2, 2), 0.0);
188         
189         //Constructor using matrix with wrong size exception
190         m = new Matrix(DIAC_ROWS, DIAC_COLS + 1);
191         diac = null;
192         try {
193             diac = new DualImageOfAbsoluteConic(m);
194             fail("IllegalArgumentException expected but not thrown");
195         } catch (IllegalArgumentException ignore) { }
196         assertNull(diac);
197         
198         //Constructor using non-symmetric matrix
199         m = new Matrix(DIAC_ROWS, DIAC_COLS);
200         m.setElementAt(0, 0, a);
201         m.setElementAt(0, 1, b);
202         m.setElementAt(0, 2, d);
203         m.setElementAt(1, 0, b + 1.0);
204         m.setElementAt(1, 1, c);
205         m.setElementAt(1, 2, e + 1.0);
206         m.setElementAt(2, 0, d + 1.0);
207         m.setElementAt(2, 1, e);
208         m.setElementAt(2, 2, f);
209         
210         diac = null;
211         try {
212             diac = new DualImageOfAbsoluteConic(m);
213             fail("NonSymmetricMatrixException expected but not thrown");
214         } catch (NonSymmetricMatrixException ignore) { }
215         assertNull(diac);                        
216     }
