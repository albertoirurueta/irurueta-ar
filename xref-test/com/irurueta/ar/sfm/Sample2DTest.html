148     @Test
149     void testSerializeDeserialize() throws WrongSizeException, IOException, ClassNotFoundException {
150         final var s1 = new Sample2D();
151 
152         // set new values
153         s1.setId("id");
154         s1.setViewId(1);
155         final var point = new InhomogeneousPoint2D(1.0, 2.0);
156         s1.setPoint(point);
157         final var reconstructedPoint = new ReconstructedPoint3D();
158         reconstructedPoint.setPoint(new InhomogeneousPoint3D(1.0, 2.0, 3.0));
159         reconstructedPoint.setId("2");
160         reconstructedPoint.setInlier(true);
161         s1.setReconstructedPoint(reconstructedPoint);
162         s1.setQualityScore(1.0);
163         final var cov = Matrix.identity(2, 2);
164         s1.setCovariance(cov);
165         final var data = new CustomPointColorData();
166         data.setId("3");
167         data.setQualityScore(1.0);
168         s1.setColorData(data);
169 
170         // check
171         assertEquals("id", s1.getId());
172         assertEquals(1, s1.getViewId());
173         assertSame(point, s1.getPoint());
174         assertSame(reconstructedPoint, s1.getReconstructedPoint());
175         assertEquals(1.0, s1.getQualityScore(), 0.0);
176         assertSame(cov, s1.getCovariance());
177         assertSame(data, s1.getColorData());
178 
179         // serialize and deserialize
180         final var bytes = SerializationHelper.serialize(s1);
181         final var s2 = SerializationHelper.<Sample2D>deserialize(bytes);
182 
183         // check
184         assertEquals(s1.getId(), s2.getId());
185         assertEquals(s1.getViewId(), s2.getViewId());
186         assertEquals(s1.getPoint(), s2.getPoint());
187         assertEquals(s1.getReconstructedPoint().getPoint(), s2.getReconstructedPoint().getPoint());
188         assertEquals(s1.getReconstructedPoint().getId(), s2.getReconstructedPoint().getId());
189         assertEquals(s1.getReconstructedPoint().isInlier(), s2.getReconstructedPoint().isInlier());
190         assertEquals(s1.getQualityScore(), s2.getQualityScore(), 0.0);
191         assertEquals(s1.getCovariance(), s2.getCovariance());
192         assertEquals(s1.getColorData().getId(), s2.getColorData().getId());
193         assertEquals(s1.getColorData().getQualityScore(), s2.getColorData().getQualityScore(), 0.0);
194     }
