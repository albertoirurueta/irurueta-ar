424     @Test
425     public void testTriangulate() throws LockedException, NotReadyException, Point3DTriangulationException {
426         int numValid = 0;
427         for (int t = 0; t < TIMES; t++) {
428             final UniformRandomizer randomizer = new UniformRandomizer(new Random());
429 
430             // obtain number of views
431             final int numViews = randomizer.nextInt(MIN_VIEWS, MAX_VIEWS);
432 
433             // create a random 3D point
434             final Point3D point3D = new InhomogeneousPoint3D(
435                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
436                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
437                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
438 
439             final List<Point2D> points2D = new ArrayList<>();
440             final List<PinholeCamera> cameras = new ArrayList<>();
441             final Point3D previousCameraCenter = new InhomogeneousPoint3D();
442             final double[] weights = new double[numViews];
443             for (int i = 0; i < numViews; i++) {
444                 // create a random camera
445                 final double horizontalFocalLength = randomizer.nextDouble(MIN_FOCAL_LENGTH, MAX_FOCAL_LENGTH);
446                 final double verticalFocalLength = randomizer.nextDouble(MIN_FOCAL_LENGTH, MAX_FOCAL_LENGTH);
447                 final double skewness = randomizer.nextDouble(MIN_SKEWNESS, MAX_SKEWNESS);
448                 final double horizontalPrincipalPoint = randomizer.nextDouble(
449                         MIN_PRINCIPAL_POINT, MAX_PRINCIPAL_POINT);
450                 final double verticalPrincipalPoint = randomizer.nextDouble(
451                         MIN_PRINCIPAL_POINT, MAX_PRINCIPAL_POINT);
452 
453                 final double alphaEuler = randomizer.nextDouble(MIN_ANGLE_DEGREES,
454                         MAX_ANGLE_DEGREES) * Math.PI / 180.0;
455                 final double betaEuler = randomizer.nextDouble(MIN_ANGLE_DEGREES,
456                         MAX_ANGLE_DEGREES) * Math.PI / 180.0;
457                 final double gammaEuler = randomizer.nextDouble(MIN_ANGLE_DEGREES,
458                         MAX_ANGLE_DEGREES) * Math.PI / 180.0;
459 
460                 final double cameraSeparationX = randomizer.nextDouble(
461                         MIN_CAMERA_SEPARATION, MAX_CAMERA_SEPARATION);
462                 final double cameraSeparationY = randomizer.nextDouble(
463                         MIN_CAMERA_SEPARATION, MAX_CAMERA_SEPARATION);
464                 final double cameraSeparationZ = randomizer.nextDouble(
465                         MIN_CAMERA_SEPARATION, MAX_CAMERA_SEPARATION);
466 
467                 weights[i] = randomizer.nextDouble(MIN_WEIGHT, MAX_WEIGHT);
468 
469                 final PinholeCameraIntrinsicParameters intrinsic = new PinholeCameraIntrinsicParameters(
470                         horizontalFocalLength, verticalFocalLength,
471                         horizontalPrincipalPoint, verticalPrincipalPoint, skewness);
472 
473                 final MatrixRotation3D rotation = new MatrixRotation3D(alphaEuler, betaEuler, gammaEuler);
474 
475                 final Point3D cameraCenter = new InhomogeneousPoint3D(
476                         previousCameraCenter.getInhomX() + cameraSeparationX,
477                         previousCameraCenter.getInhomY() + cameraSeparationY,
478                         previousCameraCenter.getInhomZ() + cameraSeparationZ);
479 
480                 final PinholeCamera camera = new PinholeCamera(intrinsic, rotation, cameraCenter);
481 
482                 // project 3D point using camera
483                 final Point2D point2D = camera.project(point3D);
484 
485                 cameras.add(camera);
486                 points2D.add(point2D);
487             }
488 
489             // create triangulator
490             final WeightedHomogeneousSinglePoint3DTriangulator triangulator =
491                     new WeightedHomogeneousSinglePoint3DTriangulator(points2D, cameras, weights, this);
492 
493             // check default values
494             assertTrue(triangulator.isReady());
495             assertFalse(triangulator.isLocked());
496             assertEquals(0, triangulateStart);
497             assertEquals(0, triangulateEnd);
498 
499             final Point3D triangulated = triangulator.triangulate();
500 
501             // check correctness
502             assertTrue(triangulator.isReady());
503             assertFalse(triangulator.isLocked());
504             assertEquals(1, triangulateStart);
505             assertEquals(1, triangulateEnd);
506             reset();
507 
508             if (point3D.distanceTo(triangulated) > ABSOLUTE_ERROR) {
509                 continue;
510             }
511             assertEquals(0.0, point3D.distanceTo(triangulated), ABSOLUTE_ERROR);
512 
513             numValid++;
514             break;
515         }
516 
517         assertTrue(numValid > 0);
518     }

520     @Override
521     public void onTriangulateStart(final SinglePoint3DTriangulator triangulator) {
522         triangulateStart++;
523         checkLocked((WeightedHomogeneousSinglePoint3DTriangulator) triangulator);
524     }
525 
526     @Override
527     public void onTriangulateEnd(final SinglePoint3DTriangulator triangulator) {
528         triangulateEnd++;
529         checkLocked((WeightedHomogeneousSinglePoint3DTriangulator) triangulator);
530     }
531 
532     private void reset() {
533         triangulateStart = triangulateEnd = 0;
534     }
535 
536     private void checkLocked(
537             final WeightedHomogeneousSinglePoint3DTriangulator triangulator) {
538         try {
539             triangulator.setListener(this);
540             fail("LockedException expected but not thrown");
541         } catch (final LockedException ignore) {
542         }
543         try {
544             triangulator.setPointsAndCameras(null, null);
545             fail("LockedException expected but not thrown");
546         } catch (final LockedException ignore) {
547         }
548         try {
549             triangulator.setPointsCamerasAndWeights(null, null, null);
550             fail("LockedException expected but not thrown");
551         } catch (final LockedException ignore) {
552         }
553         try {
554             triangulator.setMaxCorrespondences(10);
555             fail("LockedException expected but not thrown");
556         } catch (final LockedException ignore) {
557         }
558         try {
559             triangulator.setSortWeightsEnabled(true);
560             fail("LockedException expected but not thrown");
561         } catch (final LockedException ignore) {
562         }
563         try {
564             triangulator.triangulate();
565             fail("LockedException expected but not thrown");
566         } catch (final LockedException ignore) {
567         } catch (final Exception e) {
568             fail("LockedException expected but not thrown");
569         }
570         assertTrue(triangulator.isLocked());
571     }
