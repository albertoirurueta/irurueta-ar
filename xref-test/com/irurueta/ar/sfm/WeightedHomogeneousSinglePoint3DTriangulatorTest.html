333     @Test
334     void testTriangulate() throws LockedException, NotReadyException, Point3DTriangulationException {
335         var numValid = 0;
336         for (var t = 0; t < TIMES; t++) {
337             final var randomizer = new UniformRandomizer();
338 
339             // obtain number of views
340             final var numViews = randomizer.nextInt(MIN_VIEWS, MAX_VIEWS);
341 
342             // create a random 3D point
343             final var point3D = new InhomogeneousPoint3D(
344                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
345                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
346                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
347 
348             final var points2D = new ArrayList<Point2D>();
349             final var cameras = new ArrayList<PinholeCamera>();
350             final var previousCameraCenter = new InhomogeneousPoint3D();
351             final var weights = new double[numViews];
352             for (var i = 0; i < numViews; i++) {
353                 // create a random camera
354                 final var horizontalFocalLength = randomizer.nextDouble(MIN_FOCAL_LENGTH, MAX_FOCAL_LENGTH);
355                 final var verticalFocalLength = randomizer.nextDouble(MIN_FOCAL_LENGTH, MAX_FOCAL_LENGTH);
356                 final var skewness = randomizer.nextDouble(MIN_SKEWNESS, MAX_SKEWNESS);
357                 final var horizontalPrincipalPoint = randomizer.nextDouble(MIN_PRINCIPAL_POINT, MAX_PRINCIPAL_POINT);
358                 final var verticalPrincipalPoint = randomizer.nextDouble(MIN_PRINCIPAL_POINT, MAX_PRINCIPAL_POINT);
359 
360                 final var alphaEuler = Math.toRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
361                 final var betaEuler = Math.toRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
362                 final var gammaEuler = Math.toRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
363 
364                 final var cameraSeparationX = randomizer.nextDouble(MIN_CAMERA_SEPARATION, MAX_CAMERA_SEPARATION);
365                 final var cameraSeparationY = randomizer.nextDouble(MIN_CAMERA_SEPARATION, MAX_CAMERA_SEPARATION);
366                 final var cameraSeparationZ = randomizer.nextDouble(MIN_CAMERA_SEPARATION, MAX_CAMERA_SEPARATION);
367 
368                 weights[i] = randomizer.nextDouble(MIN_WEIGHT, MAX_WEIGHT);
369 
370                 final var intrinsic = new PinholeCameraIntrinsicParameters(horizontalFocalLength, verticalFocalLength,
371                         horizontalPrincipalPoint, verticalPrincipalPoint, skewness);
372 
373                 final var rotation = new MatrixRotation3D(alphaEuler, betaEuler, gammaEuler);
374 
375                 final var cameraCenter = new InhomogeneousPoint3D(
376                         previousCameraCenter.getInhomX() + cameraSeparationX,
377                         previousCameraCenter.getInhomY() + cameraSeparationY,
378                         previousCameraCenter.getInhomZ() + cameraSeparationZ);
379 
380                 final var camera = new PinholeCamera(intrinsic, rotation, cameraCenter);
381 
382                 // project 3D point using camera
383                 final var point2D = camera.project(point3D);
384 
385                 cameras.add(camera);
386                 points2D.add(point2D);
387             }
388 
389             // create triangulator
390             final var triangulator = new WeightedHomogeneousSinglePoint3DTriangulator(points2D, cameras, weights,
391                     this);
392 
393             // check default values
394             assertTrue(triangulator.isReady());
395             assertFalse(triangulator.isLocked());
396             assertEquals(0, triangulateStart);
397             assertEquals(0, triangulateEnd);
398 
399             final var triangulated = triangulator.triangulate();
400 
401             // check correctness
402             assertTrue(triangulator.isReady());
403             assertFalse(triangulator.isLocked());
404             assertEquals(1, triangulateStart);
405             assertEquals(1, triangulateEnd);
406             reset();
407 
408             if (point3D.distanceTo(triangulated) > ABSOLUTE_ERROR) {
409                 continue;
410             }
411             assertEquals(0.0, point3D.distanceTo(triangulated), ABSOLUTE_ERROR);
412 
413             numValid++;
414             break;
415         }
416 
417         assertTrue(numValid > 0);
418     }

420     @Override
421     public void onTriangulateStart(final SinglePoint3DTriangulator triangulator) {
422         triangulateStart++;
423         checkLocked((WeightedHomogeneousSinglePoint3DTriangulator) triangulator);
424     }
425 
426     @Override
427     public void onTriangulateEnd(final SinglePoint3DTriangulator triangulator) {
428         triangulateEnd++;
429         checkLocked((WeightedHomogeneousSinglePoint3DTriangulator) triangulator);
430     }
431 
432     private void reset() {
433         triangulateStart = triangulateEnd = 0;
434     }
435 
436     private void checkLocked(final WeightedHomogeneousSinglePoint3DTriangulator triangulator) {
437         assertThrows(LockedException.class, () -> triangulator.setListener(this));
438         assertThrows(LockedException.class, () -> triangulator.setPointsAndCameras(null, null));
439         assertThrows(LockedException.class,
440                 () -> triangulator.setPointsCamerasAndWeights(null, null, null));
441         assertThrows(LockedException.class, () -> triangulator.setMaxCorrespondences(10));
442         assertThrows(LockedException.class, () -> triangulator.setSortWeightsEnabled(true));
443         assertThrows(LockedException.class, triangulator::triangulate);
444         assertTrue(triangulator.isLocked());
445     }
