169     @Test
170     void testSerializeDeserialize() throws IOException, ClassNotFoundException {
171         final var cfg1 = new KnownBaselineTwoViewsSparseReconstructorConfiguration();
172 
173         // set new values
174         cfg1.setNonRobustFundamentalMatrixEstimatorMethod(FundamentalMatrixEstimatorMethod.SEVEN_POINTS_ALGORITHM);
175         cfg1.setRobustFundamentalMatrixEstimatorMethod(RobustEstimatorMethod.RANSAC);
176         cfg1.setFundamentalMatrixRefined(true);
177         cfg1.setFundamentalMatrixCovarianceKept(true);
178         cfg1.setFundamentalMatrixConfidence(0.8);
179         cfg1.setFundamentalMatrixMaxIterations(500);
180         cfg1.setFundamentalMatrixThreshold(1.0);
181         cfg1.setFundamentalMatrixComputeAndKeepInliers(false);
182         cfg1.setFundamentalMatrixComputeAndKeepResiduals(false);
183         cfg1.setInitialCamerasEstimatorMethod(InitialCamerasEstimatorMethod.DUAL_ABSOLUTE_QUADRIC);
184         cfg1.setDaqUseHomogeneousPointTriangulator(false);
185         cfg1.setInitialCamerasAspectRatio(0.9);
186         cfg1.setPrincipalPointX(0.1);
187         cfg1.setPrincipalPointY(-0.1);
188         cfg1.setInitialCamerasCorrectorType(CorrectorType.GOLD_STANDARD);
189         cfg1.setInitialCamerasMarkValidTriangulatedPoints(false);
190         final var intrinsic1a = new PinholeCameraIntrinsicParameters();
191         cfg1.setInitialIntrinsic1(intrinsic1a);
192         final var intrinsic1b = new PinholeCameraIntrinsicParameters();
193         cfg1.setInitialIntrinsic2(intrinsic1b);
194         cfg1.setGeneralSceneAllowed(false);
195         cfg1.setPlanarSceneAllowed(false);
196         cfg1.setRobustPlanarHomographyEstimatorMethod(RobustEstimatorMethod.RANSAC);
197         cfg1.setPlanarHomographyRefined(false);
198         cfg1.setPlanarHomographyCovarianceKept(true);
199         cfg1.setPlanarHomographyConfidence(0.8);
200         cfg1.setPlanarHomographyMaxIterations(500);
201         cfg1.setPlanarHomographyThreshold(1e-3);
202         cfg1.setPlanarHomographyComputeAndKeepInliers(false);
203         cfg1.setPlanarHomographyComputeAndKeepResiduals(false);
204         cfg1.setPlanarHomographyComputeAndKeepResiduals(false);
205         cfg1.setBaseline(0.5);
206 
207         // check
208         assertEquals(FundamentalMatrixEstimatorMethod.SEVEN_POINTS_ALGORITHM,
209                 cfg1.getNonRobustFundamentalMatrixEstimatorMethod());
210         assertEquals(RobustEstimatorMethod.RANSAC, cfg1.getRobustFundamentalMatrixEstimatorMethod());
211         assertTrue(cfg1.isFundamentalMatrixRefined());
212         assertTrue(cfg1.isFundamentalMatrixCovarianceKept());
213         assertEquals(0.8, cfg1.getFundamentalMatrixConfidence(), 0.0);
214         assertEquals(500, cfg1.getFundamentalMatrixMaxIterations());
215         assertEquals(1.0, cfg1.getFundamentalMatrixThreshold(), 0.0);
216         assertFalse(cfg1.getFundamentalMatrixComputeAndKeepInliers());
217         assertFalse(cfg1.getFundamentalMatrixComputeAndKeepResiduals());
218         assertEquals(InitialCamerasEstimatorMethod.DUAL_ABSOLUTE_QUADRIC, cfg1.getInitialCamerasEstimatorMethod());
219         assertFalse(cfg1.getDaqUseHomogeneousPointTriangulator());
220         assertEquals(0.9, cfg1.getInitialCamerasAspectRatio(), 0.0);
221         assertEquals(0.1, cfg1.getPrincipalPointX(), 0.0);
222         assertEquals(-0.1, cfg1.getPrincipalPointY(), 0.0);
223         assertFalse(cfg1.getInitialCamerasMarkValidTriangulatedPoints());
224         assertSame(intrinsic1a, cfg1.getInitialIntrinsic1());
225         assertSame(intrinsic1b, cfg1.getInitialIntrinsic2());
226         assertFalse(cfg1.isGeneralSceneAllowed());
227         assertFalse(cfg1.isPlanarSceneAllowed());
228         assertEquals(RobustEstimatorMethod.RANSAC, cfg1.getRobustPlanarHomographyEstimatorMethod());
229         assertFalse(cfg1.isPlanarHomographyRefined());
230         assertTrue(cfg1.isPlanarHomographyCovarianceKept());
231         assertEquals(0.8, cfg1.getPlanarHomographyConfidence(), 0.0);
232         assertEquals(500, cfg1.getPlanarHomographyMaxIterations());
233         assertEquals(1e-3, cfg1.getPlanarHomographyThreshold(), 0.0);
234         assertFalse(cfg1.getPlanarHomographyComputeAndKeepInliers());
235         assertFalse(cfg1.getPlanarHomographyComputeAndKeepResiduals());
236         assertFalse(cfg1.getPlanarHomographyComputeAndKeepResiduals());
237         assertEquals(0.5, cfg1.getBaseline(), 0.0);
238 
239         // serialize and deserialize
240         final var bytes = SerializationHelper.serialize(cfg1);
241         final var cfg2 = SerializationHelper.<KnownBaselineTwoViewsSparseReconstructorConfiguration>deserialize(bytes);
242 
243         // check
244         assertEquals(cfg1.getNonRobustFundamentalMatrixEstimatorMethod(),
245                 cfg2.getNonRobustFundamentalMatrixEstimatorMethod());
246         assertEquals(cfg1.getRobustFundamentalMatrixEstimatorMethod(),
247                 cfg2.getRobustFundamentalMatrixEstimatorMethod());
248         assertEquals(cfg1.isFundamentalMatrixRefined(), cfg2.isFundamentalMatrixRefined());
249         assertEquals(cfg1.isFundamentalMatrixCovarianceKept(), cfg2.isFundamentalMatrixCovarianceKept());
250         assertEquals(cfg1.getFundamentalMatrixConfidence(), cfg2.getFundamentalMatrixConfidence(), 0.0);
251         assertEquals(cfg1.getFundamentalMatrixMaxIterations(), cfg2.getFundamentalMatrixMaxIterations());
252         assertEquals(cfg1.getFundamentalMatrixThreshold(), cfg2.getFundamentalMatrixThreshold(), 0.0);
253         assertEquals(cfg1.getFundamentalMatrixComputeAndKeepInliers(),
254                 cfg2.getFundamentalMatrixComputeAndKeepInliers());
255         assertEquals(cfg1.getFundamentalMatrixComputeAndKeepResiduals(),
256                 cfg2.getFundamentalMatrixComputeAndKeepResiduals());
257         assertEquals(cfg1.getInitialCamerasEstimatorMethod(), cfg2.getInitialCamerasEstimatorMethod());
258         assertEquals(cfg1.getDaqUseHomogeneousPointTriangulator(), cfg2.getDaqUseHomogeneousPointTriangulator());
259         assertEquals(cfg1.getInitialCamerasAspectRatio(), cfg2.getInitialCamerasAspectRatio(), 0.0);
260         assertEquals(cfg1.getPrincipalPointX(), cfg2.getPrincipalPointX(), 0.0);
261         assertEquals(cfg1.getPrincipalPointY(), cfg2.getPrincipalPointY(), 0.0);
262         assertEquals(cfg1.getInitialCamerasMarkValidTriangulatedPoints(),
263                 cfg2.getInitialCamerasMarkValidTriangulatedPoints());
264         assertEquals(cfg1.getInitialIntrinsic1().getInternalMatrix(), cfg2.getInitialIntrinsic1().getInternalMatrix());
265         assertEquals(cfg1.getInitialIntrinsic2().getInternalMatrix(), cfg2.getInitialIntrinsic2().getInternalMatrix());
266         assertEquals(cfg1.isGeneralSceneAllowed(), cfg2.isGeneralSceneAllowed());
267         assertEquals(cfg1.isPlanarSceneAllowed(), cfg2.isPlanarSceneAllowed());
268         assertEquals(cfg1.getRobustPlanarHomographyEstimatorMethod(), cfg2.getRobustPlanarHomographyEstimatorMethod());
269         assertEquals(cfg1.isPlanarHomographyRefined(), cfg2.isPlanarHomographyRefined());
270         assertEquals(cfg1.isPlanarHomographyCovarianceKept(), cfg2.isPlanarHomographyCovarianceKept());
271         assertEquals(cfg1.getPlanarHomographyConfidence(), cfg2.getPlanarHomographyConfidence(), 0.0);
272         assertEquals(cfg1.getPlanarHomographyMaxIterations(), cfg2.getPlanarHomographyMaxIterations());
273         assertEquals(cfg1.getPlanarHomographyThreshold(), cfg2.getPlanarHomographyThreshold(), 0.0);
274         assertEquals(cfg1.getPlanarHomographyComputeAndKeepInliers(), cfg2.getPlanarHomographyComputeAndKeepInliers());
275         assertEquals(cfg1.getPlanarHomographyComputeAndKeepResiduals(),
276                 cfg2.getPlanarHomographyComputeAndKeepResiduals());
277         assertEquals(cfg1.getPlanarHomographyComputeAndKeepResiduals(),
278                 cfg2.getPlanarHomographyComputeAndKeepResiduals());
279         assertEquals(cfg1.getBaseline(), cfg2.getBaseline(), 0.0);
280     }
