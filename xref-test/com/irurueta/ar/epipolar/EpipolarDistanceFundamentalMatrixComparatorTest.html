591     private FundamentalMatrix createRandomFundamentalMatrix() throws AlgebraException,
592             InvalidFundamentalMatrixException {
593         FundamentalMatrix fundamentalMatrix;
594         int rank;
595         do {
596             var internalMatrix = Matrix.createWithUniformRandomValues(FundamentalMatrix.FUNDAMENTAL_MATRIX_ROWS,
597                     FundamentalMatrix.FUNDAMENTAL_MATRIX_COLS, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
598 
599             // ensure that internal matrix has rank 2
600             final var decomposer = new SingularValueDecomposer(internalMatrix);
601             decomposer.decompose();
602 
603             // if rank is less than 2 we need to
604             // pick another random matrix
605             rank = decomposer.getRank();
606 
607             final var u = decomposer.getU();
608             final var w = decomposer.getW();
609             final var v = decomposer.getV();
610             final var transV = v.transposeAndReturnNew();
611 
612             // set last element to 0 to force rank 2
613             w.setElementAt(2, 2, 0.0);
614 
615             internalMatrix = u.multiplyAndReturnNew(w.multiplyAndReturnNew(transV));
616 
617             fundamentalMatrix = new FundamentalMatrix(internalMatrix);
618         } while (rank < 2);
619 
620         return fundamentalMatrix;
621     }
