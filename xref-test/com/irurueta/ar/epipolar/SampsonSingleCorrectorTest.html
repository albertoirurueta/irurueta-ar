38      private static final double ABSOLUTE_ERROR = 1e-6;
39  
40      private static final double MIN_PROJECTED_ERROR = -1.0;
41      private static final double MAX_PROJECTED_ERROR = 1.0;
42  
43      private static final double MIN_RANDOM_VALUE = -100.0;
44      private static final double MAX_RANDOM_VALUE = 100.0;
45      private static final double MIN_FOCAL_LENGTH = 1.0;
46      private static final double MAX_FOCAL_LENGTH = 100.0;
47      private static final double MIN_SKEWNESS = -1.0;
48      private static final double MAX_SKEWNESS = 1.0;
49      private static final double MIN_PRINCIPAL_POINT = 0.0;
50      private static final double MAX_PRINCIPAL_POINT = 100.0;
51      private static final double MIN_ANGLE_DEGREES = -30.0;
52      private static final double MAX_ANGLE_DEGREES = 30.0;
53      private static final double MIN_CAMERA_SEPARATION = 5.0;
54      private static final double MAX_CAMERA_SEPARATION = 10.0;
55  
56      private static final int TIMES = 100;

58      @Test
59      void testConstructor() throws WrongSizeException, NotAvailableException, NotReadyException, LockedException,
60              DecomposerException, InvalidFundamentalMatrixException {
61          // test constructor without arguments
62          var corrector = new SampsonSingleCorrector();
63  
64          // check default values
65          assertNull(corrector.getLeftPoint());
66          assertNull(corrector.getRightPoint());
67          assertNull(corrector.getFundamentalMatrix());
68          assertFalse(corrector.isReady());
69          assertNull(corrector.getLeftCorrectedPoint());
70          assertNull(corrector.getRightCorrectedPoint());
71          assertEquals(CorrectorType.SAMPSON_CORRECTOR, corrector.getType());
72  
73          // test constructor with fundamental matrix
74          final var emptyFundamentalMatrix = new FundamentalMatrix();
75          corrector = new SampsonSingleCorrector(emptyFundamentalMatrix);
76  
77          // check default values
78          assertNull(corrector.getLeftPoint());
79          assertNull(corrector.getRightPoint());
80          assertSame(emptyFundamentalMatrix, corrector.getFundamentalMatrix());
81          assertFalse(corrector.isReady());
82          assertNull(corrector.getLeftCorrectedPoint());
83          assertNull(corrector.getRightCorrectedPoint());
84          assertEquals(CorrectorType.SAMPSON_CORRECTOR, corrector.getType());
85  
86          // test constructor with left and right points
87          final var leftPoint = Point2D.create();
88          final var rightPoint = Point2D.create();
89          corrector = new SampsonSingleCorrector(leftPoint, rightPoint);
90  
91          // check default values
92          assertSame(leftPoint, corrector.getLeftPoint());
93          assertSame(rightPoint, corrector.getRightPoint());
94          assertNull(corrector.getFundamentalMatrix());
95          assertFalse(corrector.isReady());
96          assertNull(corrector.getLeftCorrectedPoint());
97          assertNull(corrector.getRightCorrectedPoint());
98          assertEquals(CorrectorType.SAMPSON_CORRECTOR, corrector.getType());
99  
100         // test constructor with left and right points and a valid fundamental
101         // matrix
102         FundamentalMatrix fundamentalMatrix;
103         int rank;
104         do {
105             var internalMatrix = Matrix.createWithUniformRandomValues(FundamentalMatrix.FUNDAMENTAL_MATRIX_ROWS,
106                     FundamentalMatrix.FUNDAMENTAL_MATRIX_COLS, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
107 
108             // ensure that internal matrix has rank 2
109             final var decomposer = new SingularValueDecomposer(internalMatrix);
110             decomposer.decompose();
111 
112             // if rank is less than 2 we need to
113             // pick another random matrix
114             rank = decomposer.getRank();
115 
116             final var u = decomposer.getU();
117             final var w = decomposer.getW();
118             final var v = decomposer.getV();
119             final var transV = v.transposeAndReturnNew();
120 
121             // set last element to 0 to force rank 2
122             w.setElementAt(2, 2, 0.0);
123 
124             internalMatrix = u.multiplyAndReturnNew(w.multiplyAndReturnNew(transV));
125 
126             fundamentalMatrix = new FundamentalMatrix(internalMatrix);
127         } while (rank < 2);
128 
129         corrector = new SampsonSingleCorrector(leftPoint, rightPoint, fundamentalMatrix);
130 
131         // check default values
132         assertSame(leftPoint, corrector.getLeftPoint());
133         assertSame(rightPoint, corrector.getRightPoint());
134         assertSame(fundamentalMatrix, corrector.getFundamentalMatrix());
135         assertTrue(corrector.isReady());
136         assertNull(corrector.getLeftCorrectedPoint());
137         assertNull(corrector.getRightCorrectedPoint());
138         assertEquals(CorrectorType.SAMPSON_CORRECTOR, corrector.getType());
139 
140         // if we use a fundamental matrix without internal matrix defined, then
141         // corrector is not ready
142         corrector = new SampsonSingleCorrector(leftPoint, rightPoint, emptyFundamentalMatrix);
143 
144         // check default values
145         assertSame(leftPoint, corrector.getLeftPoint());
146         assertSame(rightPoint, corrector.getRightPoint());
147         assertSame(emptyFundamentalMatrix, corrector.getFundamentalMatrix());
148         assertFalse(corrector.isReady());
149         assertNull(corrector.getLeftCorrectedPoint());
150         assertNull(corrector.getRightCorrectedPoint());
151         assertEquals(CorrectorType.SAMPSON_CORRECTOR, corrector.getType());
152     }

210     @Test
211     void testCorrect() throws InvalidPairOfCamerasException, com.irurueta.geometry.estimators.NotReadyException {
212         var numImproved = 0;
213         for (var t = 0; t < TIMES; t++) {
214             // create intrinsic parameters
215             final var randomizer = new UniformRandomizer();
216             final var alphaEuler1 = 0.0;
217             final var betaEuler1 = 0.0;
218             final var gammaEuler1 = 0.0;
219             final var alphaEuler2 = Math.toRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
220             final var betaEuler2 = Math.toRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
221             final var gammaEuler2 = Math.toRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
222 
223             final var horizontalFocalLength1 = randomizer.nextDouble(MIN_FOCAL_LENGTH, MAX_FOCAL_LENGTH);
224             final var verticalFocalLength1 = randomizer.nextDouble(MIN_FOCAL_LENGTH, MAX_FOCAL_LENGTH);
225             final var horizontalFocalLength2 = randomizer.nextDouble(MIN_FOCAL_LENGTH, MAX_FOCAL_LENGTH);
226             final var verticalFocalLength2 = randomizer.nextDouble(MIN_FOCAL_LENGTH, MAX_FOCAL_LENGTH);
227 
228             final var skewness1 = randomizer.nextDouble(MIN_SKEWNESS, MAX_SKEWNESS);
229             final var skewness2 = randomizer.nextDouble(MIN_SKEWNESS, MAX_SKEWNESS);
230 
231             final var horizontalPrincipalPoint1 = randomizer.nextDouble(MIN_PRINCIPAL_POINT, MAX_PRINCIPAL_POINT);
232             final var verticalPrincipalPoint1 = randomizer.nextDouble(MIN_PRINCIPAL_POINT, MAX_PRINCIPAL_POINT);
233             final var horizontalPrincipalPoint2 = randomizer.nextDouble(MIN_PRINCIPAL_POINT, MAX_PRINCIPAL_POINT);
234             final var verticalPrincipalPoint2 = randomizer.nextDouble(MIN_PRINCIPAL_POINT, MAX_PRINCIPAL_POINT);
235 
236             final var cameraSeparation = randomizer.nextDouble(MIN_CAMERA_SEPARATION, MAX_CAMERA_SEPARATION);
237 
238             final var intrinsic1 = new PinholeCameraIntrinsicParameters(horizontalFocalLength1, verticalFocalLength1,
239                     horizontalPrincipalPoint1, verticalPrincipalPoint1, skewness1);
240             final var intrinsic2 = new PinholeCameraIntrinsicParameters(horizontalFocalLength2, verticalFocalLength2,
241                     horizontalPrincipalPoint2, verticalPrincipalPoint2, skewness2);
242 
243             // camera centers
244             final var cameraCenter1 = new InhomogeneousPoint3D(randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
245                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
246                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
247             final var cameraCenter2 = new InhomogeneousPoint3D(cameraCenter1.getInhomX() + cameraSeparation,
248                     cameraCenter1.getInhomY() + cameraSeparation, cameraCenter1.getInhomZ() + cameraSeparation);
249 
250             final var rotation1 = new MatrixRotation3D(alphaEuler1, betaEuler1, gammaEuler1);
251             final var rotation2 = new MatrixRotation3D(alphaEuler2, betaEuler2, gammaEuler2);
252 
253             // create random 3D point to project
254             final var pointToProject = new InhomogeneousPoint3D(
255                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
256                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
257                     randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
258 
259             // create two cameras
260             final var camera1 = new PinholeCamera(intrinsic1, rotation1, cameraCenter1);
261             final var camera2 = new PinholeCamera(intrinsic2, rotation2, cameraCenter2);
262 
263             // project 3D point with both cameras
264             final var leftPoint = camera1.project(pointToProject);
265             final var rightPoint = camera2.project(pointToProject);
266 
267             // add error to projected points
268             final var errorLeftX = randomizer.nextDouble(MIN_PROJECTED_ERROR, MAX_PROJECTED_ERROR);
269             final var errorLeftY = randomizer.nextDouble(MIN_PROJECTED_ERROR, MAX_PROJECTED_ERROR);
270             final var errorRightX = randomizer.nextDouble(MIN_PROJECTED_ERROR, MAX_PROJECTED_ERROR);
271             final var errorRightY = randomizer.nextDouble(MIN_PROJECTED_ERROR, MAX_PROJECTED_ERROR);
272 
273             final var wrongLeftPoint = new HomogeneousPoint2D(leftPoint.getInhomX() + errorLeftX,
274                     leftPoint.getInhomY() + errorLeftY, 1.0);
275             final var wrongRightPoint = new HomogeneousPoint2D(rightPoint.getInhomX() + errorRightX,
276                     rightPoint.getInhomY() + errorRightY, 1.0);
277 
278             // create fundamental matrix for the same pair of cameras used to
279             // project point
280             final var fundamentalMatrix = new FundamentalMatrix(camera1, camera2);
281 
282             // check that points without error belong to epipolar lines
283             var rightEpipolarLine = fundamentalMatrix.getRightEpipolarLine(leftPoint);
284             var leftEpipolarLine = fundamentalMatrix.getLeftEpipolarLine(rightPoint);
285             assertTrue(rightEpipolarLine.isLocus(rightPoint, ABSOLUTE_ERROR));
286             assertTrue(leftEpipolarLine.isLocus(leftPoint, ABSOLUTE_ERROR));
287 
288             fundamentalMatrix.normalize();
289 
290             // use corrector to fix points with error
291             final var corrector = new SampsonSingleCorrector(wrongLeftPoint, wrongRightPoint, fundamentalMatrix);
292 
293             assertTrue(corrector.isReady());
294 
295             corrector.correct();
296 
297             final var correctedLeftPoint = corrector.getLeftCorrectedPoint();
298             final var correctedRightPoint = corrector.getRightCorrectedPoint();
299 
300             rightEpipolarLine = fundamentalMatrix.getRightEpipolarLine(correctedLeftPoint);
301             leftEpipolarLine = fundamentalMatrix.getLeftEpipolarLine(correctedRightPoint);
302 
303             final var correctedDistanceLeft = leftEpipolarLine.signedDistance(correctedLeftPoint);
304             final var correctedDistanceRight = rightEpipolarLine.signedDistance(correctedRightPoint);
305 
306             rightEpipolarLine = fundamentalMatrix.getRightEpipolarLine(wrongLeftPoint);
307             leftEpipolarLine = fundamentalMatrix.getLeftEpipolarLine(wrongRightPoint);
308 
309             final var wrongDistanceLeft = leftEpipolarLine.signedDistance(wrongLeftPoint);
310             final var wrongDistanceRight = rightEpipolarLine.signedDistance(wrongRightPoint);
311 
312             // check that corrector has indeed reduced the amount of projection
313             // error
314             if (Math.abs(correctedDistanceLeft) > Math.abs(wrongDistanceLeft)) {
315                 continue;
316             }
317             assertTrue(Math.abs(correctedDistanceLeft) <= Math.abs(wrongDistanceLeft));
318             if (Math.abs(correctedDistanceRight) > Math.abs(wrongDistanceRight)) {
319                 continue;
320             }
321             assertTrue(Math.abs(correctedDistanceRight) <= Math.abs(wrongDistanceRight));
322 
323             numImproved++;
324             break;
325         }
326         assertTrue(numImproved > 0);
327     }
