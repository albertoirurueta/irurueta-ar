265     @Test
266     public void testCompare() throws InvalidPairOfCamerasException,
267             NotReadyException, LockedException, 
268             FundamentalMatrixComparatorException {
269         UniformRandomizer randomizer = new UniformRandomizer(new Random());
270         double alphaEuler1 = 0.0;
271         double betaEuler1 = 0.0;
272         double gammaEuler1 = 0.0;
273         double alphaEuler2 = randomizer.nextDouble(MIN_ANGLE_DEGREES, 
274                 MAX_ANGLE_DEGREES) * Math.PI / 180.0;
275         double betaEuler2 = randomizer.nextDouble(MIN_ANGLE_DEGREES, 
276                 MAX_ANGLE_DEGREES) * Math.PI / 180.0;
277         double gammaEuler2 = randomizer.nextDouble(MIN_ANGLE_DEGREES, 
278                 MAX_ANGLE_DEGREES) * Math.PI / 180.0;
279         
280         double horizontalFocalLength1 = randomizer.nextDouble(MIN_FOCAL_LENGTH,
281                 MAX_FOCAL_LENGTH);
282         double verticalFocalLength1 = randomizer.nextDouble(MIN_FOCAL_LENGTH, 
283                 MAX_FOCAL_LENGTH);
284         double horizontalFocalLength2 = randomizer.nextDouble(MIN_FOCAL_LENGTH,
285                 MAX_FOCAL_LENGTH);
286         double verticalFocalLength2 = randomizer.nextDouble(MIN_FOCAL_LENGTH, 
287                 MAX_FOCAL_LENGTH);
288         
289         double skewness1 = randomizer.nextDouble(MIN_SKEWNESS, MAX_SKEWNESS);
290         double skewness2 = randomizer.nextDouble(MIN_SKEWNESS, MAX_SKEWNESS);
291         
292         double horizontalPrincipalPoint1 = randomizer.nextDouble(
293                 MIN_PRINCIPAL_POINT, MAX_PRINCIPAL_POINT);
294         double verticalPrincipalPoint1 = randomizer.nextDouble(
295                 MIN_PRINCIPAL_POINT, MAX_PRINCIPAL_POINT);
296         double horizontalPrincipalPoint2 = randomizer.nextDouble(
297                 MIN_PRINCIPAL_POINT, MAX_PRINCIPAL_POINT);
298         double verticalPrincipalPoint2 = randomizer.nextDouble(
299                 MIN_PRINCIPAL_POINT, MAX_PRINCIPAL_POINT);
300         
301         Point3D cameraCenter1 = new InhomogeneousPoint3D(
302                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
303                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
304                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
305         Point3D cameraCenter2 = new InhomogeneousPoint3D(
306                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
307                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE),
308                 randomizer.nextDouble(MIN_RANDOM_VALUE, MAX_RANDOM_VALUE));
309         
310         Rotation3D rotation1 = new MatrixRotation3D(alphaEuler1, betaEuler1, 
311                 gammaEuler1);
312         Rotation3D rotation2 = new MatrixRotation3D(alphaEuler2, betaEuler2, 
313                 gammaEuler2);
314         
315         PinholeCameraIntrinsicParameters intrinsic1 = 
316                 new PinholeCameraIntrinsicParameters(horizontalFocalLength1, 
317                 verticalFocalLength1, horizontalPrincipalPoint1, 
318                 verticalPrincipalPoint1, skewness1);
319         PinholeCameraIntrinsicParameters intrinsic2 = 
320                 new PinholeCameraIntrinsicParameters(horizontalFocalLength2, 
321                 verticalFocalLength2, horizontalPrincipalPoint2, 
322                 verticalPrincipalPoint2, skewness2);
323         
324         PinholeCamera camera1 = new PinholeCamera(intrinsic1, rotation1, 
325                 cameraCenter1);
326         PinholeCamera camera2 = new PinholeCamera(intrinsic2, rotation2,
327                 cameraCenter2);
328         
329         FundamentalMatrix fundamentalMatrix1 = new FundamentalMatrix(camera1, 
330                 camera2);
331         FundamentalMatrix fundamentalMatrix2 = new FundamentalMatrix(camera1, 
332                 camera2);
333         
334         AlgebraicFundamentalMatrixComparator comparator = 
335                 new AlgebraicFundamentalMatrixComparator(fundamentalMatrix1, 
336                 fundamentalMatrix2, this);
337         
338         //check status
339         assertFalse(comparator.isLocked());
340         assertTrue(comparator.isReady());
341         assertEquals(compareStart, 0);
342         assertEquals(compareEnd, 0);
343         
344         //compare
345         assertEquals(comparator.compare(), 0.0, ABSOLUTE_ERROR);
346         
347         //check correctness
348         assertTrue(comparator.isReady());
349         assertFalse(comparator.isLocked());
350         assertEquals(compareStart, 1);
351         assertEquals(compareEnd, 1);
352         reset();
353     }

356     public void onCompareStart(FundamentalMatrixComparator comparator) {
357         compareStart++;
358         checkLocked((AlgebraicFundamentalMatrixComparator)comparator);
359     }

362     public void onCompareEnd(FundamentalMatrixComparator comparator) {

365     }

373     private void reset() {
374         compareStart = compareEnd = 0;

377     private void checkLocked(AlgebraicFundamentalMatrixComparator comparator){
378         assertTrue(comparator.isLocked());
379         try {
380             comparator.setGroundTruthFundamentalMatrix(null);
381             fail("LockedException expected but not thrown");
382         } catch (LockedException ignore) { }
383         try {
384             comparator.setOtherFundamentalMatrix(null);
385             fail("LockedException expected but not thrown");
386         } catch (LockedException ignore) { }
387         try {
388             comparator.setListener(null);
389             fail("LockedException expected but not thrown");
390         } catch (LockedException ignore) { }
391         try {
392             comparator.compare();
393             fail("LockedException expected but not thrown");
394         } catch (LockedException ignore) {
395         } catch (Exception ignore) {
396             fail("LockedException expected but not thrown");
397         }
398     }

400     private FundamentalMatrix createRandomFundamentalMatrix() 
401             throws AlgebraException, InvalidFundamentalMatrixException {
402         FundamentalMatrix fundamentalMatrix;
403         int rank;
404         do {
405             Matrix internalMatrix = Matrix.createWithUniformRandomValues(
406                     FundamentalMatrix.FUNDAMENTAL_MATRIX_ROWS, 
407                     FundamentalMatrix.FUNDAMENTAL_MATRIX_COLS, MIN_RANDOM_VALUE,
408                     MAX_RANDOM_VALUE);
409             
410             //ensure that internal matrix has rank 2
411             SingularValueDecomposer decomposer = new SingularValueDecomposer(
412                     internalMatrix);
413             decomposer.decompose();
414             
415             rank = decomposer.getRank(); //if rank is less than 2 we need to
416                                          //pick another random matrix
417             
418             Matrix U = decomposer.getU();
419             Matrix W = decomposer.getW();
420             Matrix V = decomposer.getV();
421             Matrix transV = V.transposeAndReturnNew();
422             
423             //set last element to 0 to force rank 2
424             W.setElementAt(2, 2, 0.0);
425             
426             internalMatrix = U.multiplyAndReturnNew(W.multiplyAndReturnNew(
427                     transV));
428             
429             fundamentalMatrix = new FundamentalMatrix(internalMatrix);
430         } while (rank < 2);
431         
432         return fundamentalMatrix;
433     }
